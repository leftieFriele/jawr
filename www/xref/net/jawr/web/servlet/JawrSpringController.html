14  package net.jawr.web.servlet;
15  
16  import java.util.HashMap;
17  import java.util.Map;
18  import java.util.Properties;
19  
20  import javax.servlet.ServletContext;
21  import javax.servlet.http.HttpServletRequest;
22  import javax.servlet.http.HttpServletResponse;
23  
24  import net.jawr.web.resource.bundle.factory.util.PathNormalizer;
25  
26  import org.apache.log4j.Logger;
27  import org.springframework.beans.factory.InitializingBean;
28  import org.springframework.web.context.ServletContextAware;
29  import org.springframework.web.servlet.ModelAndView;
30  import org.springframework.web.servlet.mvc.Controller;
31  import org.springframework.web.util.UrlPathHelper;
32  
33  /***
34   * A Spring Controller implementation which uses a JawrRequestHandler instance to provide 
35   * with Jawr functionality within a Spring DispatcherServlet instance. 
36   * 
37   * @author Jordi Hernández Sellés
38   *
39   */
40  public class JawrSpringController implements Controller, ServletContextAware, InitializingBean {
41  	
42  	private JawrRequestHandler requestHandler;
43  	private Map initParams;
44  	// Init params
45  	private String type;
46  	private String configPropertiesSourceClass;
47  	private String mapping;
48  	private String controllerMapping;
49  	private final UrlPathHelper helper = new UrlPathHelper();
50  	
51  	// Config
52  	private Properties configuration;
53  	
54  	private ServletContext context;
55  	private static final Logger log = Logger.getLogger(JawrSpringController.class);
56  
57  	
58  
59  	/* (non-Javadoc)
60  	 * @see org.springframework.web.servlet.mvc.Controller#handleRequest(javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
61  	 */
62  	public ModelAndView handleRequest(HttpServletRequest request, HttpServletResponse response) throws Exception {
63  		
64  		String requestedPath = helper.getPathWithinServletMapping(request);
65  		
66  		if(null != controllerMapping)
67  			requestedPath = request.getPathInfo().substring(controllerMapping.length());
68  		
69  		requestHandler.processRequest(requestedPath, request, response);
70  		return null;
71  	}
72  
73  	/* (non-Javadoc)
74  	 * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
75  	 */
76  	public void afterPropertiesSet() throws Exception {
77  		initParams = new HashMap(3);
78  		initParams.put("type",type);
79  		initParams.put("configPropertiesSourceClass",configPropertiesSourceClass);
80  		String fullMapping = "";
81  		if(null != mapping)
82  			fullMapping = mapping;
83  		if(null != controllerMapping)
84  			fullMapping = PathNormalizer.joinPaths(fullMapping, controllerMapping);
85  		
86  		initParams.put("mapping",fullMapping);
87  		if(log.isDebugEnabled())
88  			log.debug("Initializing Jawr Controller's JawrRequestHandler");
89  		requestHandler = new JawrRequestHandler(context,initParams,configuration);
90  	}
91  
92  	/* (non-Javadoc)
93  	 * @see org.springframework.web.context.ServletContextAware#setServletContext(javax.servlet.ServletContext)
94  	 */
95  	public void setServletContext(ServletContext context) {
96  		this.context = context;
97  	}
98  
99  	/***
100 	 * @param type the type to set
101 	 */
102 	public void setType(String type) {
103 		this.type = type;
104 	}
105 
106 	/***
107 	 * @param configPropertiesSourceClass the configPropertiesSourceClass to set
108 	 */
109 	public void setConfigPropertiesSourceClass(String configPropertiesSourceClass) {
110 		this.configPropertiesSourceClass = configPropertiesSourceClass;
111 	}
112 
113 	/***
114 	 * @param mapping the mapping to set
115 	 */
116 	public void setMapping(String mapping) {
117 		this.mapping = mapping;
118 	}
119 
120 	/***
121 	 * @param configuration the configuration to set
122 	 */
123 	public void setConfiguration(Properties configuration) {
124 		this.configuration = configuration;
125 	}
126 
127 	/***
128 	 * @param controllerMapping the controllerMapping to set
129 	 */
130 	public void setControllerMapping(String controllerMapping) {
131 		if(controllerMapping.endsWith("/"))
132 			controllerMapping = controllerMapping.substring(0,controllerMapping.length()-1);
133 		this.controllerMapping = controllerMapping;
134 	}
