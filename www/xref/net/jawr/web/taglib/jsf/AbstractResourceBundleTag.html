18  import javax.faces.component.UIOutput;
19  import javax.faces.context.FacesContext;

21  import javax.servlet.http.HttpServletRequest;
22  
23  import net.jawr.web.resource.bundle.renderer.BundleRenderer;
24  import net.jawr.web.servlet.RendererRequestUtils;

26  /***
27   * Abstract implementation of a facelets taglib component which will retrieve a Jawr config
28   * object from the servlet context and use it to render bundles of resources according 
29   * to its src attribute.  
30   * 
31   * @author Jordi Hernández Sellés
32   *
33   */
34  public abstract class AbstractResourceBundleTag extends UIOutput {
35  
36  	protected BundleRenderer renderer;
37  	protected boolean useRandomParam = true;    
38  	
39  	/* (non-Javadoc)
40  	 * @see javax.faces.component.UIComponentBase#encodeBegin(javax.faces.context.FacesContext)
41  	 */
42  	public void encodeBegin(FacesContext context) throws IOException {
43  		
44  		// Initialize attributes
45  		String useRandom = (String)getAttributes().get("useRandomParam");
46          if(null != useRandom)
47          	this.useRandomParam = Boolean.valueOf(useRandom).booleanValue();
48          
49          String src = (String)getAttributes().get("src"); 
50          
51          // src is mandatory
52          if(null == src)
53          	throw new IllegalStateException("The src attribute is mandatory for this Jawr tag. ");
54          
55          // Get an instance of the renderer. 
56          if(null == this.renderer || !this.renderer.getBundler().getConfig().isValid())
57               this.renderer = createRenderer(context);		
58          
59          ResponseWriter writer = context.getResponseWriter();
60          HttpServletRequest request = ((HttpServletRequest)context.getExternalContext().getRequest());
61          String localeKey = this.renderer.getBundler().getConfig().getLocaleResolver().resolveLocaleCode(request);
62         
63          // Determine if gzip is feasible for the current client. 
64          boolean isGzippable = RendererRequestUtils.isRequestGzippable(request,renderer.getBundler().getConfig());
65           
66          renderer.renderBundleLinks( src,
67                                       request.getContextPath(),
68                                       localeKey,
69                                       RendererRequestUtils.getAddedBundlesLog(request),
70                                       isGzippable,
71                                       writer);
72  
73  		
74  		super.encodeBegin(context);
75  	}
76  	
77      
78  	/***
79  	 * Retrieve the ResourceCollector from context. Each implementation will use a different key
80  	 * to retrieve it. 
81  	 * @return
82  	 */
83  	protected abstract BundleRenderer createRenderer(FacesContext context);
84  }
