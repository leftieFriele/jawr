23  /***
24   * Creates a Jason like structure (an object literal) from a set of message resource properties. 
25   * 
26   * @author Jordi Hernández Sellés
27   */
28  public class BundleStringJasonifier {
29  	private Map keyMap;
30  	private Properties bundleValues;
31  	
32  	private static final String FUNC = "p(";
33  	
34  	
35  
36  	public BundleStringJasonifier(Properties bundleValues) {
37  		super();
38  		this.bundleValues = bundleValues;
39  		this.keyMap = new HashMap();
40  		Enumeration keys = this.bundleValues.keys();
41  		
42  		// Create map tree with all the message keys. 
43  		while(keys.hasMoreElements())
44  			processKey((String)keys.nextElement());
45  	}
46  
47  	/***
48  	 * Creates a tree map structure from the message bundle. 
49  	 * Each messages key is split by the separator (.) and used as a
50  	 * key in the map, for whose value a new map is created, and so forth until all 
51  	 * keys form the tree structure. 
52  	 * For instance, 
53  	 * com.example.message=foo 
54  	 * com.example.somethingelse=baz 
55  	 * results in 
56  	 * [com [example [message,somethingelse] ] ]
57  	 * 
58  	 * @param key A key in the message bundle. 
59  	 */
60  	private void processKey(String key) {
61  		StringTokenizer tk = new StringTokenizer(key,".");
62  		Map currentMap = this.keyMap;
63  		while(tk.hasMoreTokens()) {
64  			String token = tk.nextToken();
65  			if(!currentMap.containsKey(token))
66  				currentMap.put(token, new HashMap());
67  			currentMap = (Map) currentMap.get(token);
68  		}
69  	}

71  	/***
72  	 * Creates a javascript object literal representing a set of message resources. 
73  	 * 
74  	 * @return StringBuffer the object literal. 
75  	 */
76  	public StringBuffer serializeBundles() {
77  		StringBuffer sb = new StringBuffer("{");
78  		
79  		// Iterates over the 
80  		for(Iterator it = keyMap.keySet().iterator(); it.hasNext();) {
81  			String currentKey = (String) it.next();
82  			handleKey(sb,keyMap,currentKey,currentKey,!it.hasNext());
83  		}
84  		
85  		return sb.append("}");
86  	}
87  	
88  	/***
89  	 * Processes a leaf from the key map, adding its name and values recursively in a javascript object literal structure, 
90  	 * where values are invocations of a method that returns a function. 
91  	 * 
92  	 * @param sb Stringbuffer to append the javascript code. 
93  	 * @param currentLeaf Current Map from the keys tree. 
94  	 * @param currentKey Current key from the keys tree. 
95  	 * @param fullKey Key with ancestors as it appears in the message bundle(foo --> com.mycompany.foo)
96  	 * @param isLeafLast Wether this is the las item in the current leaf, to append a separator. 
97  	 */
98  	private void handleKey( final StringBuffer sb, 
99  							Map currentLeaf, 
100 							String currentKey, 
101 							String fullKey, 
102 							boolean isLeafLast) {
103 		
104 		Map newLeaf = (Map) currentLeaf.get(currentKey);
105 		
106 		if(bundleValues.containsKey(fullKey)) {
107 			addValuedKey(sb,currentKey,fullKey);
108 			if(!newLeaf.isEmpty()) {
109 				sb.append(",({");
110 				for(Iterator it = newLeaf.keySet().iterator(); it.hasNext();) {
111 					String newKey = (String) it.next();
112 					handleKey(sb,newLeaf,newKey,fullKey + "." + newKey,!it.hasNext());
113 				}
114 				sb.append("}))");
115 			}
116 			else {
117 				sb.append(")");
118 			}
119 		}
120 		else if(!newLeaf.isEmpty()) {
121 			sb.append(quote(currentKey))
122 			  .append(":{");
123 			for(Iterator it = newLeaf.keySet().iterator(); it.hasNext();) {
124 				String newKey = (String) it.next();
125 				handleKey(sb,newLeaf,newKey,fullKey + "." + newKey,!it.hasNext());
126 			}
127 			sb.append("}");
128 
129 		}
130 		if(!isLeafLast)
131 			sb.append(",");
132 	}

134 	/***
135 	 * Add a key and its value to the object literal. 
136 	 * @param sb
137 	 * @param key
138 	 * @param fullKey
139 	 */
140 	private void addValuedKey(final StringBuffer sb, String key, String fullKey) {
141 		sb	.append(quote(key))
142 		.append(":")
143 		.append(FUNC)
144 		.append(quote(bundleValues.get(fullKey).toString()));
145 	}

162 	 * Produce a string in double quotes with backslash sequences in all the
163      * right places. A backslash will be inserted within </, allowing JSON
164      * text to be delivered in HTML. In JSON text, a string cannot contain a
165      * control character or an unescaped quote or backslash.
166      * @param string A String
167      * @return  A String correctly formatted for insertion in a JSON text.
168 	 */
169 	private String quote(String string) {
170         if (string == null || string.length() == 0) {
171             return "\"\"";
172         }
173 
174         char         b;
175         char         c = 0;
176         int          i;
177         int          len = string.length();
178         StringBuffer sb = new StringBuffer(len + 4);
179         String       t;
180 
181         sb.append('"');
182         for (i = 0; i < len; i += 1) {
183             b = c;
184             c = string.charAt(i);
185             switch (c) {
186             case '//':
187             case '"':
188                 sb.append('//');
189                 sb.append(c);
190                 break;
191             case '/':
192                 if (b == '<') {
193                     sb.append('//');
194                 }
195                 sb.append(c);
196                 break;
197             case '\b':
198                 sb.append("//b");
199                 break;
200             case '\t':
201                 sb.append("//t");
202                 break;
203             case '\n':
204                 sb.append("//n");
205                 break;
206             case '\f':
207                 sb.append("//f");
208                 break;
209             case '\r':
210                 sb.append("//r");
211                 break;
212             default:
213                 if (c < ' ' || (c >= '\u0080' && c < '\u00a0') ||
214                                (c >= '\u2000' && c < '\u2100')) {
215                     t = "000" + Integer.toHexString(c);
216                     sb.append("//u" + t.substring(t.length() - 4));
217                 } else {
218                     sb.append(c);
219                 }
220             }
221         }
222         sb.append('"');
223         return sb.toString();
224     }
