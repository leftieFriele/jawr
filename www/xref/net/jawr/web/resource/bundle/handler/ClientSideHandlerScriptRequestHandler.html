32  /***
33   * Handles requests for the client side script used in non dynamic html pages. 
34   * 
35   * @author Jordi Hernández Sellés
36   */
37  public class ClientSideHandlerScriptRequestHandler {
38  	private ResourceBundlesHandler rsHandler;
39  	private JawrConfig config;
40  	private Map handlerCache;
41  	private static final long startTime = System.currentTimeMillis();
42  
43  	
44  	/***
45  	 * Placeholder for a script stringbuffer and its hashcode, meant to 
46  	 * avoid constant recalculation of the hash value. 
47  	 * 
48  	 * @author Jordi Hernández Sellés
49  	 */
50  	private class Handler {
51  		String hash;
52  		StringBuffer data;
53  		Handler(StringBuffer data, String hash){
54  			this.data = data;
55  			this.hash = hash;
56  		}
57  	}
58  	
59  	public ClientSideHandlerScriptRequestHandler(
60  			ResourceBundlesHandler rsHandler, JawrConfig config) {
61  		super();
62  		this.rsHandler = rsHandler;
63  		this.config = config;
64  		this.handlerCache = ConcurrentCollectionsFactory.buildConcurrentHashMap();
65  	}
66  	
67  
68  	/***
69  	 * Generates a locale dependent script used to include bundles in non dynamic html pages. 
70  	 * Uses a cache of said scripts to avoid constant regeneration. 
71  	 * It also keeps track of eTags and if-modified-since headers to take advantage of 
72  	 * client side caching. 
73  	 * 
74  	 * @param request
75  	 * @param response
76  	 */
77  	public void handleClientSideHandlerRequest(HttpServletRequest request, HttpServletResponse response){
78  		Handler handler;
79  		String locale = config.getLocaleResolver().resolveLocaleCode(request);
80  		locale = null == locale ? "_d" : locale;
81  		
82  
83  		if(handlerCache.containsKey(locale)){
84  			handler = (Handler) handlerCache.get(locale);
85  		}
86  		else {	
87  			StringBuffer sb = rsHandler.getClientSideHandler().getClientSideHandlerScript(request);
88  			handler = new Handler(sb, Integer.toString(sb.hashCode()));
89  			handlerCache.put(locale, handler);
90  		}
91  
92  		// Decide wether to set a 304 response		
93  		if(useNotModifiedHeader(request,handler.hash)){
94  			response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
95  			return;
96  		}
97  		response.setHeader(HttpConstants.HEADER_ETAG, handler.hash);
98  		response.setDateHeader(HttpConstants.HEADER_LAST_MODIFIED,startTime);
99  		
100 		if(RendererRequestUtils.isRequestGzippable(request, this.config)) {			
101 			try {
102 				response.setHeader("Content-Encoding", "gzip");
103 				GZIPOutputStream gzOut = new GZIPOutputStream(response.getOutputStream());
104 				byte[] data = handler.data.toString().getBytes(this.config.getResourceCharset().name());
105 				gzOut.write(data, 0, data.length);
106 				gzOut.close();
107 			} catch (IOException e) {
108 				throw new RuntimeException("Unexpected IOException writing ClientSideHandlerScript",e);
109 			}
110 		}
111 		else {
112 			StringReader rd = new StringReader(handler.data.toString());
113 			try {
114 				Writer writer = response.getWriter();
115                 IOUtils.copy(rd, writer);
116 				rd.close();
117 				writer.close();
118 			} catch (IOException e) {
119 				throw new RuntimeException("Unexpected IOException writing ClientSideHandlerScript",e);
120 			}
121 		}
122 	}
123 	
124 	/***
125 	 * Determines wether a response should get a 304 response and empty body, according to 
126 	 * etags and if-modified-since headers. 
127 	 * 
128 	 * @param request
129 	 * @param scriptEtag
130 	 * @return
131 	 */
132 	private boolean useNotModifiedHeader(HttpServletRequest request,String scriptEtag) {
133 		long modifiedHeader = -1;
134 		try
135         {
136             modifiedHeader = request.getDateHeader(HttpConstants.HEADER_IF_MODIFIED);
137 			if(modifiedHeader != -1)
138 				modifiedHeader -= modifiedHeader % 1000;
139         }
140         catch (RuntimeException ex){}
141         String eTag = request.getHeader(HttpConstants.HEADER_IF_NONE);
142         if(modifiedHeader == -1) {
143         	return scriptEtag.equals(eTag);
144         }
145         else if(null == eTag){
146         	return modifiedHeader <= startTime;
147         }
148         else {
149         	return scriptEtag.equals(eTag) && modifiedHeader <= startTime;
150         }
151         
152         
153 	}
