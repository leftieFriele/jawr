14  package net.jawr.web.resource.bundle.generator.dwr;
15  
16  import org.directwebremoting.util.IdGenerator;
17  
18  /***
19   * This class adds a small script to pages, which is needed when integrating JAWR with DWR.
20   * 
21   * 
22   * @author Jordi Hernández Sellés
23   */
24  public class DWRParamWriter {
25  	private static final IdGenerator generator = new IdGenerator();
26  	private static final int pageIdLength = 16;
27  	private static boolean useDynamicSessionId; // Should this evolve to an interceptor chain for the AbstractBundleLinkRenderer?
28  	
29  	
30  	
31  	/***
32  	 * Adds a script with DWR needed params, including a generated ID that DWR needs. 
33  	 * 
34  	 * @param contextPath
35  	 * @param dwrPath
36  	 * @return
37  	 */
38  	public static StringBuffer buildRequestSpecificParams(String contextPath,String dwrPath) {
39  		StringBuffer sb = new StringBuffer("<script type=\"text/javascript\">if(!JAWR){var JAWR = {};};");
40  		sb.append("JAWR.jawr_dwr_path='");
41          sb.append(dwrPath).append("';");
42          
43          if(useDynamicSessionId)
44          	sb.append("JAWR.dwr_scriptSessionId='").append(generator.generateId(pageIdLength)).append("';");
45  		
46          sb.append("JAWR.app_context_path='").append(contextPath).append("';");
47          sb.append("</script>").append("\n");
48  		
49  		return sb;
50  	}
51  
52  
53  
54  	public static void setUseDynamicSessionId(boolean useDynamicSessionId) {
55  		DWRParamWriter.useDynamicSessionId = useDynamicSessionId;
56  	}
57  
58  	
59  }
