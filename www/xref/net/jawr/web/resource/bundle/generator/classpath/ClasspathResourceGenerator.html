14  package net.jawr.web.resource.bundle.generator.classpath;

16  import java.io.FileNotFoundException;
17  import java.io.InputStream;
18  import java.io.Reader;
19  import java.nio.channels.Channels;
20  import java.nio.channels.ReadableByteChannel;
21  import java.nio.charset.Charset;
22  import java.util.Locale;
23  
24  import javax.servlet.ServletContext;
25  
26  import net.jawr.web.config.JawrConfig;
27  import net.jawr.web.resource.bundle.factory.util.ClassLoaderResourceUtils;
28  import net.jawr.web.resource.bundle.generator.ResourceGenerator;
29  
30  /***
31   * Loads resources from the classpath.  
32   * 
33   * @author Jordi Hernández Sellés
34   */
35  public class ClasspathResourceGenerator implements ResourceGenerator {
36  
37  	/* (non-Javadoc)
38  	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource(java.lang.String, java.nio.charset.Charset)
39  	 */
40  	public Reader createResource(String path, JawrConfig config, ServletContext servletContext,Locale locale, Charset charset) {
41  		try {
42  			InputStream is = ClassLoaderResourceUtils.getResourceAsStream(path, this);
43  			 ReadableByteChannel chan = Channels.newChannel(is);
44  			 return Channels.newReader(chan,charset.newDecoder (),-1);
45  		} catch (FileNotFoundException e) {
46  			throw new RuntimeException(e);
47  		}
48  	}
49  
50  }
