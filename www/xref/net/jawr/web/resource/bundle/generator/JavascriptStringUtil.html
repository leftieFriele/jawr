36  	 * Produce a string in double quotes with backslash sequences in all the
37  	 * right places. A backslash will be inserted within </, allowing JSON
38  	 * text to be delivered in HTML. In JSON text, a string cannot contain a
39  	 * control character or an unescaped quote or backslash.
40  	 * @param string A String
41  	 * @return  A String correctly formatted for insertion in a JSON text.
42  	 */
43  	public static String quote(String string) {
44  	    if (string == null || string.length() == 0) {
45  	        return "\"\"";
46  	    }	
47  	    int          len = string.length();
48  	    StringBuffer sb = new StringBuffer(len + 4);
49  	
50  	    sb.append('"');
51  	    sb.append(escape(string));
52  	    sb.append('"');
53  	    return sb.toString();
54  	}

56  	/***
57  	 * Escapes a string so it is valid between double quotes in javascript. 
58  	 * @param string
59  	 * @return
60  	 */
61  	public static String escape(String string) {
62  		char         b;
63  	    char         c = 0;
64  	    int          i;
65  	    int          len = string.length();
66  	    StringBuffer sb = new StringBuffer(len + 2);
67  	    String       t;
68  	    for (i = 0; i < len; i += 1) {
69  	        b = c;
70  	        c = string.charAt(i);
71  	        switch (c) {
72  	        case '//':
73  	        case '"':
74  	            sb.append('//');
75  	            sb.append(c);
76  	            break;
77  	        case '/':
78  	            if (b == '<') {
79  	                sb.append('//');
80  	            }
81  	            sb.append(c);
82  	            break;
83  	        case '\b':
84  	            sb.append("//b");
85  	            break;
86  	        case '\t':
87  	            sb.append("//t");
88  	            break;
89  	        case '\n':
90  	            sb.append("//n");
91  	            break;
92  	        case '\f':
93  	            sb.append("//f");
94  	            break;
95  	        case '\r':
96  	            sb.append("//r");
97  	            break;
98  	        default:
99  	            if (c < ' ' || (c >= '\u0080' && c < '\u00a0') ||
100 	                           (c >= '\u2000' && c < '\u2100')) {
101 	                t = "000" + Integer.toHexString(c);
102 	                sb.append("//u" + t.substring(t.length() - 4));
103 	            } else {
104 	                sb.append(c);
105 	            }
106 	        }
107 	    }
108 	    return sb.toString();
109 	}
