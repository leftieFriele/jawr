30  /***
31   * Registry for resource generators, which create scripts or CSS data dynamically, 
32   * as opposed to the usual behavior of reading a resource from the war file. 
33   * It provides methods to determine if a path mapping should be handled by a generator, 
34   * and to actually render the resource using the appropiate generator. 
35   * Path mappings which require generation will use a prefix (preferrably one which ends with 
36   * a colon, such as 'messages:'). 
37   * Generators provided with Jawr will be automatically mapped. 
38   * 
39   * @author Jordi Hernández Sellés
40   *
41   */
42  public class GeneratorRegistry {
43  	
44  	public static final String MESSAGE_BUNDLE_PREFIX = "messages";
45  	public static final String CLASSPATH_BUNDLE_PREFIX = "jar";
46  	public static final String DWR_BUNDLE_PREFIX = "dwr";
47  	public static final String COMMONS_VALIDATOR_PREFIX = "acv";
48  	
49  	public static final String PREFIX_SEPARATOR = ":";
50  	
51  	private static final Map registry = ConcurrentCollectionsFactory.buildConcurrentHashMap();
52  	private static final List prefixRegistry = ConcurrentCollectionsFactory.buildCopyOnWriteArrayList();
53  	private JawrConfig config;
54  	
55  	static
56  	{
57  		prefixRegistry.add(MESSAGE_BUNDLE_PREFIX + PREFIX_SEPARATOR);
58  		prefixRegistry.add(CLASSPATH_BUNDLE_PREFIX + PREFIX_SEPARATOR);
59  		prefixRegistry.add(DWR_BUNDLE_PREFIX + PREFIX_SEPARATOR);
60  		prefixRegistry.add(COMMONS_VALIDATOR_PREFIX + PREFIX_SEPARATOR);
61  	}
62  	
63  
64  	/***
65  	 * Use only for testing purposes.
66  	 */
67  	public GeneratorRegistry(){
68  		super();
69  	}

71  	/***
72  	 * Lazy loads generators, to avoid the need for undesired dependencies. 
73  	 * 
74  	 * @param generatorKey
75  	 * @return
76  	 */
77  	private void loadGenerator(String generatorKey) {
78  		if((MESSAGE_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
79  			registry.put(generatorKey, new ResourceBundleMessagesGenerator());
80  		}
81  		else if((CLASSPATH_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
82  			registry.put(generatorKey, new ClasspathResourceGenerator());
83  		}
84  		else if((DWR_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
85  			registry.put(generatorKey, DWRGeneratorFactory.createDWRGenerator());
86  		}
87  		else if((COMMONS_VALIDATOR_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
88  			registry.put(generatorKey, new CommonsValidatorGenerator());
89  		}
90  	}

92  	/***
93  	 * Register a generator mapping it to the specified prefix. 
94  	 * 
95  	 * @param clazz String Qualified classname of the generator
96  	 */
97  	public void registerGenerator(String clazz){
98  		ResourceGenerator generator = (ResourceGenerator) ClassLoaderResourceUtils.buildObjectInstance(clazz);
99  		
100 		if(null == generator.getMappingPrefix() || "".equals(generator.getMappingPrefix()) )
101 			throw new IllegalStateException("The getMappingPrefix() method must return something at " + clazz);
102 		
103 		String fullPrefix = generator.getMappingPrefix() + PREFIX_SEPARATOR;
104 		
105 		// Verify this prefix is unused
106 		if(prefixRegistry.contains(fullPrefix)) {
107 			String generatorName = registry.get(fullPrefix).getClass().getName();
108 			if(!clazz.equals(generatorName)) {
109 				String errorMsg = "Cannot register the generator of class " 
110 								+ generator.getClass().getName()
111 								+ " using the prefix " + generator.getMappingPrefix() + " since such prefix is being used by "
112 								+ generatorName + ". Please pecify a different return value at the getMappingPrefix() method.";
113 				throw new IllegalStateException(errorMsg);
114 			}
115 		}
116 		
117 		prefixRegistry.add(generator.getMappingPrefix() + PREFIX_SEPARATOR);
118 		registry.put(generator.getMappingPrefix() + PREFIX_SEPARATOR, generator);
119 	}

121 	/***
122 	 * Determines wether a path is to be handled by a generator. 
123 	 * @param path
124 	 * @return
125 	 */
126 	public boolean isPathGenerated(String path) {
127 		return null != matchPath(path);
128 	}

130 	/***
131 	 * Creates the contents corresponding to a path, by using the appropiate generator. 
132 	 * @param path
133 	 * @param charset
134 	 * @return
135 	 */
136 	public Reader createResource(String path) {
137 		String key = matchPath(path);
138 		Locale locale = null;
139 		
140 		if(path.indexOf('@') != -1){
141 			String localeKey = path.substring(path.indexOf('@')+1);
142 			path = path.substring(0,path.indexOf('@'));
143 			
144 			// Resourcebundle should be doing this for me...
145 			String[] params = localeKey.split("_");			
146 			switch(params.length) {
147 				case 3:
148 					locale = new Locale(params[0],params[1],params[2]);
149 					break;
150 				case 2: 
151 					locale = new Locale(params[0],params[1]);
152 					break;
153 				default:
154 					locale = new Locale(localeKey);
155 			}
156 		}
157 		GeneratorContext context = new GeneratorContext(config, path.substring(key.length()));
158 		context.setLocale(locale);
159 		
160 		return ((ResourceGenerator)registry.get(key)).createResource(context);
161 	}

164 	/***
165 	 * Returns the path to use in the generation URL for debug mode. 
166 	 * @param path
167 	 * @return
168 	 */
169 	public String getDebugModeGenerationPath(String path) {
170 		String key = matchPath(path);
171 		return ((ResourceGenerator)registry.get(key)).getDebugModeRequestPath();
172 	}

174 	/***
175 	 * Get the key from the mappings that corresponds to the specified path. 
176 	 * @param path
177 	 * @return
178 	 */
179 	private String matchPath(String path) {
180 		String rets = null;
181 		for(Iterator it = prefixRegistry.iterator();it.hasNext() && rets == null;) {
182 			String prefix = (String) it.next();
183 			if(path.startsWith(prefix))
184 				rets = prefix;			
185 		}	
186 		// Lazy load generator
187 		if(null != rets && !registry.containsKey(rets))
188 			loadGenerator(rets);
189 		
190 		return rets;
191 	}
