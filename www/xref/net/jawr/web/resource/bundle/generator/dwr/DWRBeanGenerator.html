108 	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource(java.lang.String, java.nio.charset.Charset)
109 	 */
110 	public Reader createResource(String path, Charset charset) {
111 		StringBuffer data = null;
112 		if(ENGINE_KEY.equals(path)) {
113 			data = buildEngineScript(readDWRScript(ENGINE_PATH));
114 		}
115 		else if(dwrLibraries.containsKey(path)){
116 			data = readDWRScript((String)dwrLibraries.get(path));
117 		}
118 		else if(ALL_INTERFACES_KEY.equals(path)) {
119 			data = new StringBuffer(ENGINE_INIT);
120 			data.append(getAllPublishedInterfaces());
121 		}
122 		else {
123 			data = new StringBuffer(ENGINE_INIT);
124 			StringTokenizer tk = new StringTokenizer(path,"|");
125 			while(tk.hasMoreTokens()) {
126 				data.append(getInterfaceScript(tk.nextToken()));
127 			}
128 		}
129 		
130 		return new StringReader(data.toString());
131 	}

136 	/***
137 	 * Performs replacement on the engine.js script from DWR. 
138 	 * Mainly copies what DWR does, only at startup. A couple params are actually
139 	 * replaced to references to javascript vars that jawr will create on the page. 
140 	 * 
141 	 * @param engineScript
142 	 * @return
143 	 */
144 	private StringBuffer buildEngineScript(StringBuffer engineScript) {
145 		List containers = ContainerUtil.getAllPublishedContainers(servletContext);
146 		String allowGetForSafariButMakeForgeryEasier = "";
147 		String scriptTagProtection = DwrConstants.SCRIPT_TAG_PROTECTION;
148 		String pollWithXhr = "";
149 		String sessionCookieName = "JSESSIONID";
150 		for(Iterator it = containers.iterator();it.hasNext();) {
151 			Container container = (Container) it.next();
152 			ServerLoadMonitor monitor = (ServerLoadMonitor) container.getBean(ServerLoadMonitor.class.getName());
153 			pollWithXhr = monitor.supportsStreaming() ? "false" : "true";
154 			
155 			if(null != container.getBean("allowGetForSafariButMakeForgeryEasier")){
156 				allowGetForSafariButMakeForgeryEasier = (String)container.getBean("allowGetForSafariButMakeForgeryEasier");
157 			}
158 			if(null != container.getBean("scriptTagProtection")){
159 				scriptTagProtection = (String)container.getBean("scriptTagProtection");
160 			}
161 			if(null != container.getBean("sessionCookieName")){
162 				sessionCookieName = (String)container.getBean("sessionCookieName");
163 			}			
164 		}
165 		StringBuffer sb = new StringBuffer();
166 		Matcher matcher = paramsPattern.matcher(engineScript);
167 		while(matcher.find()) {
168 			String match = matcher.group();
169 			if("${allowGetForSafariButMakeForgeryEasier}".equals(match)){
170 				matcher.appendReplacement(sb, allowGetForSafariButMakeForgeryEasier);				
171 			}
172 			else if("${pollWithXhr}".equals(match)){
173 				matcher.appendReplacement(sb, pollWithXhr);				
174 			}
175 			else if("${sessionCookieName}".equals(match)){
176 				matcher.appendReplacement(sb, sessionCookieName);				
177 			}
178 			else if("${scriptTagProtection}".equals(match)){
179 				matcher.appendReplacement(sb, scriptTagProtection);				
180 			}
181 			else if("${scriptSessionId}".equals(match)){
182 				matcher.appendReplacement(sb, "\"+JAWR.dwr_scriptSessionId+\"");				
183 			}
184 			else if("${defaultPath}".equals(match)){
185 
186 				matcher.appendReplacement(sb, "\"+JAWR.jawr_dwr_path+\"");
187 			}
188 		}
189 		matcher.appendTail(sb);
190 		return sb;
191 	}

193 	/***
194 	 * Read a DWR utils script from the classpath. 
195 	 * @param classpath
196 	 * @return
197 	 */
198 	private StringBuffer readDWRScript(String classpath) {
199 		StringBuffer sb = null;
200 		try {
201 			InputStream is = ClassLoaderResourceUtils.getResourceAsStream(classpath, this);
202 			ReadableByteChannel chan = Channels.newChannel(is);
203 			Reader r = Channels.newReader(chan,"utf-8");
204 			StringWriter sw = new StringWriter();
205 			int i = 0;
206 			while((i = r.read()) != -1) {
207 				sw.write(i);
208 			}
209 			sb = sw.getBuffer();
210 			
211 		} catch (FileNotFoundException e) {
212 			throw new RuntimeException(e);
213 		} catch (IOException e) {
214 			throw new RuntimeException(e);
215 		}
216 		
217 		return sb;
218 	}

220 	/***
221 	 * Returns a script with a specified DWR interface
222 	 * @param basePath
223 	 * @return
224 	 */
225 	private StringBuffer getInterfaceScript(String scriptName) {
226 		StringBuffer sb = new StringBuffer(ENGINE_INIT);
227 
228 		// List all containers to find all DWR interfaces
229 		List containers = ContainerUtil.getAllPublishedContainers(servletContext);
230 		boolean found = false;
231 		for(Iterator it = containers.iterator();it.hasNext() && !found;) {
232 			Container container = (Container) it.next();
233 			
234 			// The creatormanager holds the list of beans
235 			CreatorManager ctManager = (CreatorManager) container.getBean(CreatorManager.class.getName());
236 			if( null != ctManager ) {
237 				// The remoter builds interface scripts. 
238 				Remoter remoter = (Remoter) container.getBean(Remoter.class.getName());
239 				String path = getPathReplacementString(container);
240 				try {
241 					String script = remoter.generateInterfaceScript(scriptName, path);
242 					found = true;
243 					// Must remove the engine init script to avoid unneeded duplication
244 					script = removeEngineInit(script);
245 					sb.append(script);
246 				}
247 				catch(SecurityException ex){throw new RuntimeException(ex); }
248 			}
249 		}
250 		if(!found)
251 			throw new IllegalArgumentException("The DWR bean named '" + scriptName + "' was not found in any DWR configuration instance."); 
252 		
253 		return sb;
254 	}

258 	/***
259 	 * Gets the appropiate path replacement string for a DWR container
260 	 * @param container
261 	 * @return
262 	 */
263 	private String getPathReplacementString(Container container) {
264 		String path = JS_PATH_REF;
265 		if(null != container.getBean(DWR_OVERRIDEPATH_PARAM)) {
266 			path = (String) container.getBean(DWR_OVERRIDEPATH_PARAM);
267 		}
268 		else if(null != container.getBean(DWR_MAPPING_PARAM)) {
269 			path = JS_CTX_PATH + container.getBean(DWR_MAPPING_PARAM);
270 		}
271 		return path;
272 	}

274 	/***
275 	 * Returns a script with all the DWR interfaces available in the servletcontext
276 	 * @param basePath
277 	 * @return
278 	 */
279 	private StringBuffer getAllPublishedInterfaces() {
280 		
281 		StringBuffer sb = new StringBuffer();
282 
283 		// List all containers to find all DWR interfaces
284 		List containers = ContainerUtil.getAllPublishedContainers(servletContext);
285 		for(Iterator it = containers.iterator();it.hasNext();) {
286 			Container container = (Container) it.next();
287 			
288 			// The creatormanager holds the list of beans
289 			CreatorManager ctManager = (CreatorManager) container.getBean(CreatorManager.class.getName());
290 			if(null != ctManager) {
291 				// The remoter builds interface scripts. 
292 				Remoter remoter = (Remoter) container.getBean(Remoter.class.getName());
293 				
294 				String path = getPathReplacementString(container);
295 				
296 				Collection c = ctManager.getCreatorNames();
297 				for(Iterator names = c.iterator();names.hasNext();) {
298 					String script = remoter.generateInterfaceScript((String)names.next(), path);
299 					// Must remove the engine init script to avoid unneeded duplication
300 					script = removeEngineInit(script);
301 					sb.append(script);
302 				}
303 			}
304 		}
305 		return sb;
306 	}

308 	/***
309 	 * Removes the engine init script so that it is not repeated unnecesarily. 
310 	 * @param script
311 	 * @return
312 	 */
313 	private String removeEngineInit(String script) {
314 		int start = script.indexOf(ENGINE_INIT);
315 		int end = start + ENGINE_INIT.length();
316 		StringBuffer rets = new StringBuffer();
317 		
318 		if(start > 0) {
319 			rets.append(script.substring(0, start)).append("\n");
320 		}
321 		rets.append(script.substring(end));
322 		
323 		
324 		return rets.toString();
325 	}
