14  package net.jawr.web.resource.bundle.factory.mapper;

16  import java.util.ArrayList;
17  import java.util.HashMap;
18  import java.util.Iterator;
19  import java.util.List;
20  import java.util.Map;
21  
22  import net.jawr.web.exception.DuplicateBundlePathException;
23  import net.jawr.web.resource.ResourceHandler;
24  import net.jawr.web.resource.bundle.JoinableResourceBundle;
25  import net.jawr.web.resource.bundle.factory.util.PathNormalizer;
26  
27  import org.apache.log4j.Logger;

29  /***
30   * Base class to implement map-based automatic bundles generators. 
31   * The generated bundles are added to a Map instance in which the keys 
32   * are bundles ids and the values are the bundles. 
33   * 
34   * @author Jordi Hernández Sellés
35   *
36   */
37  public abstract class AbstractResourceMapper {
38  	private static final Logger log = Logger.getLogger(AbstractResourceMapper.class);
39  	protected String baseDir;
40  	protected ResourceHandler rsHandler;
41  	protected List currentBundles;
42  	protected String resourceExtension;
43  	private Map bundleMapping;
44  	
45  	public AbstractResourceMapper(String baseDir,
46  			ResourceHandler rsHandler, List currentBundles,
47  			String resourceExtension) {
48  		super();
49  		this.baseDir = PathNormalizer.normalizePath(baseDir);
50  		this.rsHandler = rsHandler;
51  		this.currentBundles = new ArrayList();
52  		if(null != currentBundles)
53  			this.currentBundles.addAll(currentBundles);
54  		this.resourceExtension = resourceExtension;
55  		this.bundleMapping = new HashMap();
56  	}
57  	
58  
59  	/***
60  	 * Find the required files to add to the mapping. Subclasses must use the addBundleToMap method. 
61  	 * @throws DuplicateBundlePathException
62  	 */
63  	protected abstract void addBundlesToMapping() throws DuplicateBundlePathException;
64  	
65  	public final Map getBundleMapping() throws DuplicateBundlePathException{
66  		addBundlesToMapping();
67  		return bundleMapping;
68  	}
69  	
70  	/***
71  	 * Add a bundle and its mapping to the resulting Map. 
72  	 * @param bundleId
73  	 * @param mapping
74  	 * @throws DuplicateBundlePathException 
75  	 */
76  	protected final void addBundleToMap(String bundleId, String mapping) throws DuplicateBundlePathException {
77  		
78  		for(Iterator it = currentBundles.iterator();it.hasNext(); ) {
79  			JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();			
80  			if(bundleId.equals(bundle.getName()) || this.bundleMapping.containsKey(bundleId)){
81  				log.fatal("Duplicate bundle id resulted from mapping:" + bundleId);
82  				throw new DuplicateBundlePathException(bundleId);
83  			}
84  		}
85  		
86  		bundleMapping.put(bundleId, mapping);
87  	}
88  	
89  	
90  }
