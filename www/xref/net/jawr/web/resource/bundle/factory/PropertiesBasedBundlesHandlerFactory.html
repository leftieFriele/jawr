92  	/***
93  	 * Create a PropertiesBasedBundlesHandlerFactory using the specified
94  	 * properties.
95  	 * 
96  	 * @param properties
97  	 * @param resourceType
98  	 *            js or css
99  	 * @param rsHandler
100 	 *            ResourceHandler to access files.
101 	 */
102 	public PropertiesBasedBundlesHandlerFactory(Properties properties,
103 			String resourceType, ResourceHandler rsHandler,
104 			GeneratorRegistry generatorRegistry) {
105 		this.props = new PropertiesConfigHelper(properties, resourceType);
106 
107 		// Create the BundlesHandlerFactory
108 		factory = new BundlesHandlerFactory();
109 		factory.setResourceHandler(rsHandler);
110 		factory.setBundlesType(resourceType);
111 
112 		// Root resources dir
113 		factory.setBaseDir(props.getProperty(RESOURCES_BASEDIR, "/"));
114 
115 		// Use cache by default
116 		factory.setUseInMemoryCache(Boolean.valueOf(
117 				props.getProperty(RESOURCES_USE_CACHE, "true")).booleanValue());
118 
119 		// Postprocessor definitions
120 		factory.setGlobalPostProcessorKeys(props
121 				.getProperty(BUNDLE_FACTORY_POSTPROCESSOR));
122 		factory.setUnitPostProcessorKeys(props
123 				.getProperty(BUNDLE_FACTORY_FILE_POSTPROCESSOR));
124 
125 		// Single or multiple bundle for orphans settings.
126 		factory.setUseSingleResourceFactory(Boolean.valueOf(
127 				props.getProperty(FACTORY_USE_SINGLE_BUNDLE, "false"))
128 				.booleanValue());
129 		factory.setSingleFileBundleName(props
130 				.getProperty(FACTORY_SINGLE_FILE_NAME));
131 
132 		// Use the automatic directory-as-bundle mapper.
133 		factory.setUseDirMapperFactory(Boolean.valueOf(
134 				props.getProperty(FACTORY_USE_DIR_MAPPER, "false"))
135 				.booleanValue());
136 		factory.setExludedDirMapperDirs(props
137 				.getPropertyAsSet(FACTORY_DIR_MAPPER_EXCLUSION));
138 
139 		// Initialize custom generators
140 		Iterator generators = props.getPropertyAsSet(CUSTOM_GENERATORS)
141 				.iterator();
142 		while (generators.hasNext()) {
143 			String generatorClass = (String) generators.next();
144 			generatorRegistry.registerGenerator(generatorClass);
145 		}
146 
147 		// Initialize custom bundles
148 		Set customBundles = new HashSet();
149 		// Check if we should use the bundle names property or
150 		// find the bundle name using the bundle id declaration :
151 		// jawr.<type>.bundle.<name>.id
152 		if(null != props.getProperty(BUNDLE_FACTORY_CUSTOM_NAMES)) {
153 			StringTokenizer tk = new StringTokenizer(props
154 					.getProperty(BUNDLE_FACTORY_CUSTOM_NAMES), ",");
155 			while (tk.hasMoreTokens()) {
156 				customBundles.add(buildCustomBundleDefinition(tk.nextToken()
157 						.trim(), false));
158 			}
159 		} else {
160 			Iterator bundleNames = props.getPropertyBundleNameSet().iterator();
161 			while (bundleNames.hasNext()) {
162 				customBundles.add(buildCustomBundleDefinition(
163 						(String) bundleNames.next(), false));
164 			}
165 		}
166 
167 		// Read custom postprocessor definitions
168 		if (null != properties.getProperty(CUSTOM_POSTPROCESSORS
169 				+ CUSTOM_POSTPROCESSORS_NAMES)) {
170 			Map customPostprocessors = new HashMap();
171 			StringTokenizer tk = new StringTokenizer(properties
172 					.getProperty(CUSTOM_POSTPROCESSORS
173 							+ CUSTOM_POSTPROCESSORS_NAMES), ",");
174 
175 			while (tk.hasMoreTokens()) {
176 				String processorKey = tk.nextToken();
177 				String processorClass = properties
178 						.getProperty(CUSTOM_POSTPROCESSORS + "." + processorKey
179 								+ CUSTOM_POSTPROCESSORS_CLASS);
180 				if (null != processorClass)
181 					customPostprocessors.put(processorKey, processorClass);
182 			}
183 			factory.setCustomPostprocessors(customPostprocessors);
184 
185 		}
186 
187 		factory.setBundleDefinitions(customBundles);
188 	}

203 	/***
204 	 * Create a BundleDefinition based on the properties file.
205 	 * 
206 	 * @param bundleName
207 	 * @return
208 	 */
209 	private ResourceBundleDefinition buildCustomBundleDefinition(
210 			String bundleName, boolean isChildBundle) {
211 
212 		// Id for the bundle
213 		String bundleId = props.getCustomBundleProperty(bundleName,
214 				BUNDLE_FACTORY_CUSTOM_ID);
215 		if (null == bundleId && !isChildBundle)
216 			throw new IllegalArgumentException(
217 					"No id defined for the bundle with name:" + bundleName
218 							+ ". Please specify one in configuration. ");
219 
220 		// Wether it's a composite or not
221 		boolean isComposite = Boolean.valueOf(
222 				props.getCustomBundleProperty(bundleName,
223 						BUNDLE_FACTORY_CUSTOM_COMPOSITE_FLAG, "false"))
224 				.booleanValue();
225 
226 		// Create definition and set its id
227 		ResourceBundleDefinition bundle = new ResourceBundleDefinition();
228 		bundle.setBundleId(bundleId);
229 
230 		// Wether it's global or not
231 		Boolean isGlobal = Boolean.valueOf(props.getCustomBundleProperty(
232 				bundleName, BUNDLE_FACTORY_CUSTOM_GLOBAL_FLAG, "false"));
233 		bundle.setGlobal(isGlobal.booleanValue());
234 
235 		// Set order if its a global bundle
236 		if (isGlobal.booleanValue()) {
237 			Integer order = Integer.valueOf(props.getCustomBundleProperty(
238 					bundleName, BUNDLE_FACTORY_CUSTOM_ORDER, "0"));
239 			bundle.setInclusionOrder(order.intValue());
240 		}
241 
242 		// Override bundle postprocessor
243 		if (null != props.getCustomBundleProperty(bundleName,
244 				BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR))
245 			bundle.setBundlePostProcessorKeys(props.getCustomBundleProperty(
246 					bundleName, BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR));
247 
248 		// Override unitary postprocessor
249 		if (null != props.getCustomBundleProperty(bundleName,
250 				BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR))
251 			bundle.setUnitaryPostProcessorKeys(props.getCustomBundleProperty(
252 					bundleName, BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR));
253 
254 		// Use only with debug mode on
255 		Boolean isDebugOnly = Boolean.valueOf(props.getCustomBundleProperty(
256 				bundleName, BUNDLE_FACTORY_CUSTOM_DEBUGONLY, "false"));
257 		bundle.setDebugOnly(isDebugOnly.booleanValue());
258 
259 		// Use only with debug mode off
260 		Boolean isDebugNever = Boolean.valueOf(props.getCustomBundleProperty(
261 				bundleName, BUNDLE_FACTORY_CUSTOM_DEBUGNEVER, "false"));
262 		bundle.setDebugNever(isDebugNever.booleanValue());
263 
264 		// Set conditional comment for IE, in case one is specified
265 		if (null != props.getCustomBundleProperty(bundleName,
266 				BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION))
267 			bundle.setIeConditionalExpression(props
268 					.getCustomBundleProperty(bundleName,
269 							BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION));
270 
271 		if (isComposite) {
272 			String childBundlesProperty = props.getCustomBundleProperty(
273 					bundleName, BUNDLE_FACTORY_CUSTOM_COMPOSITE_NAMES);
274 			if (null == childBundlesProperty)
275 				throw new IllegalArgumentException(
276 						"No child bundle names were defined for the composite bundle with name:"
277 								+ bundleName
278 								+ ". Please specify at least one in configuration. ");
279 
280 			bundle.setComposite(true);
281 
282 			// add children
283 			List children = new ArrayList();
284 			StringTokenizer tk = new StringTokenizer(childBundlesProperty, ",");
285 			while (tk.hasMoreTokens()) {
286 				ResourceBundleDefinition childDef = buildCustomBundleDefinition(
287 						tk.nextToken().trim(), true);
288 				childDef.setBundleId(bundleId);
289 				children.add(childDef);
290 			}
291 			bundle.setChildren(children);
292 		} else {
293 			String mappingsProperty = props.getCustomBundleProperty(bundleName,
294 					BUNDLE_FACTORY_CUSTOM_MAPPINGS);
295 			if (null == mappingsProperty)
296 				throw new IllegalArgumentException(
297 						"No mappings were defined for the bundle with name:"
298 								+ bundleName
299 								+ ". Please specify at least one in configuration. ");
300 
301 			// Add the mappings
302 			List mappings = new ArrayList();
303 			StringTokenizer tk = new StringTokenizer(mappingsProperty, ",");
304 			while (tk.hasMoreTokens())
305 				mappings.add(tk.nextToken().trim());
306 			bundle.setMappings(mappings);
307 
308 			String locales = props.getCustomBundleProperty(bundleName,
309 					BUNDLE_FACTORY_CUSTOM_LOCALE_VARIANTS);
310 			if (null != locales) {
311 				List localeKeys = new ArrayList();
312 				StringTokenizer tkl = new StringTokenizer(locales, ",");
313 				while (tkl.hasMoreTokens())
314 					localeKeys.add(tkl.nextToken().trim());
315 				bundle.setLocaleVariantKeys(localeKeys);
316 
317 			}
318 		}
319 
320 		return bundle;
321 	}
