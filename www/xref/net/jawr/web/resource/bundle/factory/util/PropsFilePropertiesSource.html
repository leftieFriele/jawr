14  package net.jawr.web.resource.bundle.factory.util;
15  
16  import java.io.File;
17  import java.io.FileInputStream;
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.util.Properties;
21  
22  import org.apache.log4j.Logger;
23  
24  /***
25   * ConfigPropertiesSource implementation that reads its values from a .properties file. 
26   * @author Jordi Hernández Sellés
27   */
28  public class PropsFilePropertiesSource implements ConfigPropertiesSource {
29  
30  	private static final Logger log = Logger.getLogger(PropsFilePropertiesSource.class.getName());
31  	
32  	private String configLocation;
33  	protected int propsHashCode;
34  	protected static final String FILE_PREFIX = "file:";
35  	private boolean checking;
36  	
37  	
38  	/* (non-Javadoc)
39  	 * @see net.jawr.web.resource.bundle.factory.util.ConfigPropertiesSource#getConfigProperties()
40  	 */
41  	public final Properties getConfigProperties() {		
42  		Properties props = doReadConfig();		
43  		
44  		// Initialize hashcode of newly loaded properties. 
45  		if(0 == this.propsHashCode) {
46  			this.propsHashCode = props.hashCode();
47  		}
48  		
49  		return props;
50  	}
51  
52  	/***
53  	 * Implements the loic to load the properties. Subclasses will override this method to 
54  	 * implement the specific logic to load their configuration properties. 
55  	 * @return
56  	 */
57  	protected Properties doReadConfig(){
58  		return  readConfigFile(this.configLocation);
59  	}
60  	
61  	/***
62  	 * Reads a config file at the specified path. If the path starts with file:, it will 
63  	 * be read from the filesystem. Otherwise it will be loaded from the classpath. 
64  	 * 
65  	 * @param path
66  	 * @return
67  	 */
68  	protected Properties readConfigFile(String path) {
69  		Properties props = new Properties();	
70  		// Load properties file
71  		try {	
72  			InputStream is;
73  			if(path.startsWith(FILE_PREFIX)) {
74  				if(log.isDebugEnabled() && !checking) 			
75  					log.debug("Using filesystem properties file location at: " + configLocation);
76  				is = new FileInputStream(new File(path.substring(FILE_PREFIX.length())));
77  			}
78  			else {
79  				if(log.isDebugEnabled() && !checking) 
80  					log.debug("Reading properties from file at classpath: " + configLocation);				
81  				is = ClassLoaderResourceUtils.getResourceAsStream(path,this);
82  			}
83  			// load properties into a Properties object
84  			props.load(is);
85  			
86  			
87  		} 
88  		catch (IOException e) {
89  			throw new IllegalArgumentException("jawr configuration could not be found at "
90  					+ path + ". Make sure parameter is properly set "
91  					+ "in web.xml. ");
92  		}
93  		return props;
94  	}
95  
96  	/***
97  	 * @param configLocation the configLocation to set
98  	 */
99  	public void setConfigLocation(String configLocation) {
100 		this.configLocation = configLocation;
101 	}
102 
103 	/* (non-Javadoc)
104 	 * @see net.jawr.web.resource.bundle.factory.util.ConfigPropertiesSource#configChanged()
105 	 */
106 	public final boolean configChanged() {
107 		checking = true;
108 		int currentConfigHash = doReadConfig().hashCode();
109 		boolean configChanged = this.propsHashCode != currentConfigHash;
110 		
111 		if(configChanged && log.isDebugEnabled())
112 			log.debug("Changes in configuration properties file detected.");
113 			
114 		this.propsHashCode = currentConfigHash;
115 		
116 		return configChanged;
117 	}
