114 	/***
115 	 * Detects all files that belong to this bundle and adds them to the 
116 	 * items path list. 
117 	 */
118 	private void initPathList()	{
119 		if(log.isDebugEnabled())
120 			log.debug("Creating bundle path List for " + getName());
121 		
122 		for(Iterator it = pathMappings.iterator();it.hasNext();)
123 		{
124 			String pathMapping = (String) it.next();
125 
126 			// path ends in /, the folder is included without subfolders
127 			if(pathMapping.endsWith("/")) {
128 				addItemsFromDir(pathMapping,false);
129 			}
130 			// path ends in /, the folder is included with all subfolders
131 			else if(pathMapping.endsWith("/**")){
132 				addItemsFromDir(pathMapping.substring(0,pathMapping.lastIndexOf("**")),true);
133 			}
134 			else if(pathMapping.endsWith(fileExtension)){
135 				itemPathList.add(PathNormalizer.asPath(pathMapping));
136 			}
137 			else if(pathMapping.endsWith(LICENSES_FILENAME)){
138 				licensesPathList.add(PathNormalizer.asPath(pathMapping));
139 			}
140 			else if(resourceHandler.isResourceGenerated(pathMapping)){
141 				itemPathList.add(pathMapping);
142 			}
143 			else log.warn("Wrong mapping [" + pathMapping  
144 							+ "] for bundle [" 
145 							+ this.name 
146 							+ "]. Please check configuration. ");
147 		}
148 		if(log.isDebugEnabled())
149 			log.debug("Finished creating bundle path List for " + getName());
150 	}

152 	/***
153 	 * Adds all the resources within a path to the item path list. 
154 	 * @param dirName
155 	 * @param addSubDirs boolean If subfolders will be included. In such case, 
156 	 * 							 every folder below the path is included.  
157 	 */
158 	private void addItemsFromDir(String dirName, boolean addSubDirs)
159 	{
160 		Set resources = resourceHandler.getResourceNames(dirName);
161 		
162 			
163 		if(log.isDebugEnabled()) {
164 			log.debug("Adding " + resources.size() + " resources from path [" + dirName + "] to bundle " + getName());
165 		}
166 
167 		
168 		// If the directory contains a sorting file, it is used to order the resources. 
169 		if(resources.contains(SORT_FILE_NAME) || resources.contains("/" + SORT_FILE_NAME)) {
170 			SortFileParser parser;
171 			try {
172 				String sortFilePath = PathNormalizer.joinPaths(dirName, SORT_FILE_NAME);
173 				parser = new SortFileParser(resourceHandler.getResource(sortFilePath),
174 															resources,
175 															dirName);
176 			} catch (ResourceNotFoundException e) {
177 				throw new RuntimeException("Unexpected ResourceNotFoundException when reading a sorting file[" + e.getRequestedPath() + "]",e);
178 			}
179 			List sortedResources = parser.getSortedResources();
180 			for(Iterator it = sortedResources.iterator();it.hasNext();) {
181 				String resourceName = (String) it.next();
182 				
183 				// Add subfolders or files
184 				if(resourceName.endsWith(fileExtension)){
185 					itemPathList.add(PathNormalizer.asPath(resourceName));
186 					if(log.isDebugEnabled())
187 						log.debug("Added to item path list from the sorting file:" + resourceName);
188 				}
189 				else if(addSubDirs && resourceHandler.isDirectory(resourceName))
190 					addItemsFromDir(resourceName,true);
191 			}		
192 		}
193 		
194 		// Add licenses file
195 		if(resources.contains(LICENSES_FILENAME) || resources.contains("/" + LICENSES_FILENAME)) {
196 			licensesPathList.add(PathNormalizer.joinPaths(dirName, LICENSES_FILENAME));
197 		}
198 		
199 		// Add remaining resources (remaining after sorting, or all if no sort file present)
200 		List folders = new ArrayList();
201 		for(Iterator it = resources.iterator();it.hasNext();) {
202 			String resourceName = (String) it.next();
203 			String resourcePath = PathNormalizer.joinPaths(dirName, resourceName);
204 			if(resourceName.endsWith(fileExtension)) {
205 				itemPathList.add(PathNormalizer.asPath(resourcePath));
206 
207 				if(log.isDebugEnabled())
208 					log.debug("Added to item path list:" + PathNormalizer.asPath(resourcePath));
209 			}
210 			else if(addSubDirs && resourceHandler.isDirectory(resourcePath))
211 				folders.add(resourceName);
212 		}
213 		
214 		// Add subfolders if requested. Subfolders are added last unless specified in sorting file. 
215 		if(addSubDirs)
216 		{
217 			for(Iterator it = folders.iterator();it.hasNext();) {
218 				String folderName = (String) it.next();
219 				addItemsFromDir(PathNormalizer.joinPaths(dirName, folderName),true);
220 			}
221 		}
222 	}

253 		for(Iterator it = itemPathList.iterator();it.hasNext();){
254 			String path = (String) it.next();

256 				rets.add(path + '@' + variantKey);

277     /* (non-Javadoc)
278 	* @see net.jawr.web.resource.bundle.JoinableResourceBundle#getURLPrefix()
279 	*/
280     public String getURLPrefix(String variantKey) {
281     	if(null == this.urlPrefix)
282     		throw new IllegalStateException("The bundleDataHashCode must be set before accessing the url prefix.");
283     	
284     	// Resolves the locale key like resourcebundle does
285     	if(null != variantKey && null != this.localeVariantKeys) {
286     		String key = getAvailableLocaleVariant(variantKey);
287     		if(null != key)
288     			return this.urlPrefix + "." + key + "/";
289     	}
290     	return this.urlPrefix + "/";
291     }

293     /***
294      * Resolves a registered path from a locale key, using the same algorithm used to 
295      * locate ResourceBundles. 
296      *  
297      * @param variantKey
298      * @return
299      */
300     private String getAvailableLocaleVariant(String variantKey) {
301     	String key = null;
302     	if(this.localeVariantKeys.contains(variantKey)){
303     		key = variantKey;
304     	}
305     	else {
306     		String subVar = variantKey;
307     		while(subVar.indexOf('_') != -1) {
308     			subVar = subVar.substring(0,subVar.lastIndexOf('_'));
309     			if(this.localeVariantKeys.contains(subVar)){
310         			key = subVar;
311         		}
312     		}
313     	}
314     	return key;
315     }
