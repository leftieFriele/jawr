12  package net.jawr.web.resource.bundle;
13  
14  import java.io.InputStream;
15  import java.io.InputStreamReader;
16  import java.io.IOException;
17  import java.io.Reader;
18  import java.io.Writer;
19  
20  public class IOUtils {
21  
22      // the buffer size for reading data
23      private static final int BUFFER_SIZE = 16384;
24  
25      /***
26       * Writes all the contents of a Reader to a Writer.
27       * @param reader the reader to read from
28       * @param writer the writer to write to
29       */
30      public static void copy(Reader reader, Writer writer) throws IOException {
31          char[] buf = new char[BUFFER_SIZE];
32          int num = 0;
33  
34          while ((num = reader.read(buf, 0, buf.length)) != -1) {
35              writer.write(buf, 0, num);
36          }
37      }
38  
39      /***
40       * Writes all the contents of an InputStream to a Writer.
41       * @param input the input stream to read from
42       * @param writer the writer to write to
43       */
44      public static void copy(InputStream input, Writer writer) throws IOException {
45          copy(new InputStreamReader(input), writer);
46      }
47  }
