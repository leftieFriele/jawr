14  package net.jawr.web.resource.bundle.renderer;
15  
16  import net.jawr.web.resource.bundle.handler.ResourceBundlesHandler;
17  
18  /***
19   * Renderer that creates css link tags. 
20   * 
21   * @author Jordi Hernández Sellés
22   */
23  public class CSSHTMLBundleLinkRenderer extends AbstractBundleLinkRenderer implements BundleRenderer{
24      
25      private static final String PRE_TAG = "<link rel=\"stylesheet\" type=\"text/css\" media=\"";
26      private static final String MID_TAG = "\" href=\"";
27      private static final String POST_TAG = "\" />\n";
28      private static final String POST_HTML_TAG = "\" >\n";
29      private static final String POST_XHTML_EXT_TAG = "\" ></link>\n";
30      public static final String FLAVORS_XHTML = "xhtml";
31      public static final String FLAVORS_XHTML_EXTENDED = "xhtml_ext";
32      public static final String FLAVORS_HTML = "html";
33      
34      private static String closingFlavor = POST_TAG;
35      private String media;
36      
37      /*** Creates a new instance of CSSHTMLBundleLinkRenderer */
38      public CSSHTMLBundleLinkRenderer(ResourceBundlesHandler bundler, boolean useRandomParam, String media) {
39          super(bundler, useRandomParam);
40          
41          this.media = null == media ? "screen" : media;
42      }
43      
44      /***
45       * Utility method to get the closing tag value based on 
46       * a config parameter. 
47       * @param flavor
48       * @return
49       */
50      public static void setClosingTag(String flavor) {
51      	
52      	if(FLAVORS_XHTML_EXTENDED.equalsIgnoreCase(flavor)) {
53          	closingFlavor = POST_XHTML_EXT_TAG;
54          }
55          else if(FLAVORS_HTML.equalsIgnoreCase(flavor))
56          	closingFlavor = POST_HTML_TAG;
57          else closingFlavor = POST_TAG;
58      	
59      }
60  
61      /* (non-Javadoc)
62       * @see net.jawr.web.resource.bundle.renderer.AbstractBundleLinkRenderer#createBundleLink(java.lang.String, java.lang.String)
63       */
64      protected String renderLink(String fullPath) {
65      	
66          StringBuffer sb = new StringBuffer(PRE_TAG);
67  		sb.append(media).append(MID_TAG)
68  						.append(fullPath)	
69  						.append(closingFlavor); 
70          return sb.toString();
71      }
72      
73  }
