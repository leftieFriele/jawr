30  /***
31   * Abstract base class for implementations of a link renderer. 
32   * 
33   * @author Jordi Hernández Sellés
34   */
35  public abstract class AbstractBundleLinkRenderer implements BundleRenderer {
36  	private static final String ID_SCRIPT_DWR_PATH = "__dwr_path__";
37      
38      private ResourceBundlesHandler bundler;
39      
40      private boolean useRandomParam = true;
41      
42      /*** Creates a new instance of AbstractBundleLinkRenderer
43       * @param bundler ResourceBundlesHandler Handles resolving of paths. 
44       */
45      protected AbstractBundleLinkRenderer(ResourceBundlesHandler bundler, boolean useRandomParam) {
46          this.bundler = bundler;
47          this.useRandomParam = useRandomParam;
48      }
49      
50      /***
51       * Render a link or group of links to the specified resource, included global bundles if applies. 
52       * @param requestedPath String Path that identifies a resource bundle id or one of its members. 
53       * @param contextPath String The context path to prepend to the URL. 
54       * @param includedBundles Set A set of names previously added. None of them will be written out as links to avoid duplication. 
55       * @param useGzip boolean If true, the gzip prefix is added to the URLs so the link will point to the gzipped version. 
56       * @param out Writer Writer to output the tags, typically a JSPWriter.
57       * @throws java.io.IOException 
58       */
59      public void renderBundleLinks(  String requestedPath,
60                                      String contextPath,
61                                      String variantKey,
62                                      final Set includedBundles, 
63                                      boolean useGzip, 
64                                      Writer out ) throws IOException {
65          
66      	boolean debugOn = bundler.getConfig().isDebugModeOn();
67      	JoinableResourceBundle bundle = bundler.resolveBundleForPath(requestedPath);
68  		
69      	if(null == bundle)
70  			return;
71      	
72          if( debugOn ) {
73                  addComment("Start adding members resolved by '" + requestedPath + "'. Bundle id is: '" + bundle.getName() + "'",out);
74          }
75          
76          // If DWR is being used, add a path var to the page
77          if( null != bundler.getConfig().getDwrMapping() && 
78          	includedBundles.add(ID_SCRIPT_DWR_PATH)) {
79          	StringBuffer sb = DWRParamWriter.buildRequestSpecificParams(contextPath,PathNormalizer.joinPaths(contextPath,  bundler.getConfig().getDwrMapping()));
80              out.write(sb.toString());
81          }
82  
83          // Retrieve the name or names of bundle(s) that belong to/with the requested path. 
84      	ResourceBundlePathsIterator it = bundler.getBundlePaths(bundle.getName(),
85      															new ConditionalCommentRenderer(out),
86      															variantKey);
87      	
88          // Add resources to the page as links. 
89          while(it.hasNext())
90          {
91                  String resourceName = it.nextPath();
92  
93                  // If the resource had been added before, it will not be included again. 
94                  if(includedBundles.add(resourceName)){
95                  		// In debug mode, all the resources are included separately and use a random parameter to avoid caching. 
96                  		// If useRandomParam is set to false, the links are created without the random parameter. 
97                          if( debugOn && useRandomParam && 
98                          		!bundler.getConfig().getGeneratorRegistry().isPathGenerated(resourceName)) {
99                                  int random = new Random().nextInt();
100                                 if(random < 0)
101                                 	random*=-1;
102                                 out.write(createBundleLink(resourceName + "?d=" + random,contextPath));
103                         }
104                         else if(!debugOn && useGzip)
105                             out.write(createGzipBundleLink(resourceName,contextPath));
106                         else 
107                             out.write(createBundleLink(resourceName,contextPath));
108                 }
109                 else if( debugOn ) {
110                         addComment("Skipping resource '" + resourceName +"' since it is already included in the page.",out);
111                 }
112         }
113         if( debugOn ) {
114                 addComment("Finished adding members resolved by " + requestedPath,out);
115         }
116     }

119 	/***
120      * Adds an HTML comment to the output stream. 
121      * @param commentText
122      * @param out Writer
123      * @throws IOException
124      */
125     protected final void addComment(String commentText, Writer out)  throws IOException {
126     	StringBuffer sb = new StringBuffer("<script type=\"text/javascript\">/* ");
127             sb.append(commentText).append(" */</script>").append("\n");
128             out.write(sb.toString());
129     }

131     /***
132      * Creates a link to a bundle in the page, prepending the gzip prefix to its identifier. 
133      * @param resourceName
134      * @param contextPath
135      * @return String
136      */
137     protected String createGzipBundleLink(String resourceName, String contextPath) {
138         // remove '/' from start of name
139         resourceName = resourceName.substring(1,resourceName.length());
140         return createBundleLink(BundleRenderer.GZIP_PATH_PREFIX + resourceName,contextPath);
141     }
142     
143     /***
144      * Creates a link to a bundle in the page. 
145      * @param bundleId
146      * @param contextPath
147      * @return
148      */
149     protected String createBundleLink(String bundleId, String contextPath) {
150     	
151     	// When debug mode is on and the resource is generated the path must include a parameter
152     	if( bundler.getConfig().isDebugModeOn() && 
153     		bundler.getConfig().getGeneratorRegistry().isPathGenerated(bundleId)) {
154     		try {
155 				bundleId = "generate.js?" + JawrRequestHandler.GENERATION_PARAM + "=" + URLEncoder.encode(bundleId, "UTF-8");
156 			} catch (UnsupportedEncodingException neverHappens) {/*URLEncoder:how not to use checked exceptions...*/}
157     	}
158     	String fullPath = PathNormalizer.joinPaths(bundler.getConfig().getServletMapping(), bundleId);
159     	
160     	// If context path is overriden..
161     	if(null != bundler.getConfig().getContextPathOverride()) {    		
162     		String override = bundler.getConfig().getContextPathOverride();
163     		// Blank override, create url relative to path
164     		if("".equals(override)) {
165     			fullPath = fullPath.substring(1);
166     		}
167     		else fullPath = PathNormalizer.joinPaths(override,fullPath);
168     	}    		
169     	else
170     		fullPath = PathNormalizer.joinPaths(contextPath,fullPath);
171     	
172     	return renderLink(fullPath);
173     }
174     
175     /***
176      * Creates a link to a bundle in the page, using its identifier. 
177      * @param bundleId
178      * @param contextPath
179      * @return String
180      */
181     protected abstract String renderLink(String fullPath);
182 
183     /***
184      * @return ResourceBundlesHandler The resources handler.
185      */
186     public ResourceBundlesHandler getBundler() {
187         return bundler;
188     }
