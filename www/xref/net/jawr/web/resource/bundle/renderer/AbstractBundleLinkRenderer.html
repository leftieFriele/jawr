14  package net.jawr.web.resource.bundle.renderer;
15  
16  import java.io.IOException;
17  import java.io.Writer;
18  import java.util.Iterator;
19  import java.util.List;
20  import java.util.Random;
21  import java.util.Set;
22  
23  import net.jawr.web.resource.bundle.factory.util.PathNormalizer;
24  import net.jawr.web.resource.bundle.handler.ResourceBundlesHandler;
25  
26  /***
27   * Abstract base class for implementations of a link renderer. 
28   * 
29   * @author Jordi Hernández Sellés
30   */
31  public abstract class AbstractBundleLinkRenderer implements BundleRenderer {
32      
33      private ResourceBundlesHandler bundler;
34      
35      private boolean useRandomParam = true;
36      
37      /*** Creates a new instance of AbstractBundleLinkRenderer
38       * @param bundler ResourceBundlesHandler Handles resolving of paths. 
39       */
40      protected AbstractBundleLinkRenderer(ResourceBundlesHandler bundler, boolean useRandomParam) {
41          this.bundler = bundler;
42          this.useRandomParam = useRandomParam;
43      }
44      
45      /***
46       * Render a link or group of links to the specified resource, included global bundles if applies. 
47       * @param requestedPath String Path that identifies a resource bundle id or one of its members. 
48       * @param contextPath String The context path to prepend to the URL. 
49       * @param includedBundles Set A set of names previously added. None of them will be written out as links to avoid duplication. 
50       * @param useGzip boolean If true, the gzip prefix is added to the URLs so the link will point to the gzipped version. 
51       * @param out Writer Writer to output the tags, typically a JSPWriter.
52       * @throws java.io.IOException 
53       */
54      public void renderBundleLinks(  String requestedPath,
55                                      String contextPath,
56                                      final Set includedBundles, 
57                                      boolean useGzip, 
58                                      Writer out ) throws IOException {
59          
60      	boolean debugOn = bundler.getConfig().isDebugModeOn();
61          String bundleId = bundler.resolveBundleForPath(requestedPath);
62  		
63          // Retrieve the name or names of bundle(s) that belong to/with the requested path. 
64          List resourceNames = bundler.getBundlePaths(bundleId);
65  
66          if( debugOn ) {
67                  addComment("Start adding members resolved by '" + requestedPath + "'. Bundle id is: '" + bundleId + "'",out);
68          }
69  
70          // Add resources to the page as links. 
71          for(Iterator it = resourceNames.iterator();it.hasNext();)
72          {
73                  String resourceName = (String) it.next();
74  
75                  // If the resource had been added before, it will not be included again. 
76                  if(includedBundles.add(resourceName)){
77                  		// In debug mode, all the resources are included separately and use a random parameter to avoid caching. 
78                  		// If useRandomParam is set to false, the links are created without the random parameter. 
79                          if( debugOn && useRandomParam) {
80                                  int random = new Random().nextInt();
81                                  if(random < 0)
82                                  	random*=-1;
83                                  out.write(createBundleLink(resourceName + "?d=" + random,contextPath));
84                          }
85                          else if(useGzip)
86                              out.write(createGzipBundleLink(resourceName,contextPath));
87                          else 
88                              out.write(createBundleLink(resourceName,contextPath));
89                  }
90                  else if( debugOn ) {
91                          addComment("Skipping resource '" + resourceName +"' since it is already included in the page.",out);
92                  }
93          }
94          if( debugOn ) {
95                  addComment("Finished adding members resolved by " + requestedPath,out);
96          }
97      }
98      
99

101     /***
102      * Adds an HTML comment to the output stream. 
103      * @param commentText
104      * @param out Writer
105      * @throws IOException
106      */
107     protected final void addComment(String commentText, Writer out)  throws IOException {
108     	StringBuffer sb = new StringBuffer("<!-- ");
109             sb.append(commentText).append(" -->").append("\n");
110             out.write(sb.toString());
111     }
112     
113     /***
114      * Creates a link to a bundle in the page, prepending the gzip prefix to its identifier. 
115      * @param resourceName
116      * @param contextPath
117      * @return String
118      */
119     protected String createGzipBundleLink(String resourceName, String contextPath) {
120         // remove '/' from start of name
121         resourceName = resourceName.substring(1,resourceName.length());
122         return createBundleLink(BundleRenderer.GZIP_PATH_PREFIX + resourceName,contextPath);
123     }
124     
125     protected String createBundleLink(String bundleId, String contextPath) {
126     	
127     	String fullPath = PathNormalizer.joinPaths(bundler.getConfig().getServletMapping(), bundleId);
128     	
129     	// If context path is overriden..
130     	if(null != bundler.getConfig().getContextPathOverride()) {    		
131     		String override = bundler.getConfig().getContextPathOverride();
132     		// Blank override, create url relative to path
133     		if("".equals(override)) {
134     			fullPath = fullPath.substring(1);
135     		}
136     		else fullPath = PathNormalizer.joinPaths(override,fullPath);
137     	}    		
138     	else
139     		fullPath = PathNormalizer.joinPaths(contextPath,fullPath);
140     	
141     	return renderLink(fullPath);
142     }
143     
144     /***
145      * Creates a link to a bundle in the page, using its identifier. 
146      * @param bundleId
147      * @param contextPath
148      * @return String
149      */
150     protected abstract String renderLink(String fullPath);
151 
152     /***
153      * @return ResourceBundlesHandler The resources handler.
154      */
155     public ResourceBundlesHandler getBundler() {
156         return bundler;
157     }
