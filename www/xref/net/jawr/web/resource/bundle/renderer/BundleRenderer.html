22  /***
23   * This interface defines operations to generate import tags for bundled resources. 
24   * 
25   * @author Jordi Hernández Sellés
26   */
27  public interface BundleRenderer {
28      
29      /***
30       * Prefix to append to every URL that points to a gzipped version of a resource. 
31       */
32      public static final String GZIP_PATH_PREFIX = "/gzip_";
33              
34      /***
35       * Render a link to the specified resource. 
36       * @param requestedPath String Path that identifies a resource bundle id or one of its members. 
37       * @param contextPath String The context path to prepend to the URL. 
38       * @param includedBundles Set A set of names previously added. None of them will be written out as links to avoid duplication. 
39       * @param useGzip boolean If true, the gzip prefix is added to the URLs so the link will point to the gzipped version. 
40       * @param out Writer Writer to output the tags, typically a JSPWriter. 
41       */
42      public void renderBundleLinks(  String requestedPath,
43                                      String contextPath,
44                                      final Set includedBundles, 
45                                      boolean useGzip, 
46                                      Writer out ) throws IOException;
47      
48      /***
49       * @return ResourceBundlesHandler The resources handler used by this renderer.
50       */
51      public ResourceBundlesHandler getBundler();
52  }
