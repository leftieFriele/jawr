14  package net.jawr.web.resource.bundle.postprocess.impl.yui;
15  
16  import java.io.BufferedReader;
17  import java.io.IOException;
18  import java.io.StringReader;
19  
20  import net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus;
21  
22  import org.apache.log4j.Logger;
23  import org.mozilla.javascript.ErrorReporter;
24  import org.mozilla.javascript.EvaluatorException;
25  
26  /***
27   * Implementation of Rhino's ErrorReporter, used as a callback to log errors encuntered while 
28   * parsing a javascript file. 
29   * 
30   * @author Jordi Hernández Sellés
31   */
32  public class YUIErrorReporter implements ErrorReporter {
33  	
34  	private static final Logger log = Logger.getLogger(YUIErrorReporter.class);
35  	private BundleProcessingStatus status;
36  	private StringBuffer bundleData;
37  	private int errorLine;
38  	private String yuiErrorMessage;
39  
40  	/***
41  	 * @param status Current bundling status. 
42  	 * @param bundleData Contents of the bundle, used when an error occurs to display the conlifcting line. 
43  	 */
44  	public YUIErrorReporter(BundleProcessingStatus status,StringBuffer bundleData) {
45  		super();
46  		this.status = status;
47  		this.bundleData = bundleData;
48  	}
49  
50  
51  
52  	/* (non-Javadoc)
53  	 * @see org.mozilla.javascript.ErrorReporter#error(java.lang.String, java.lang.String, int, java.lang.String, int)
54  	 */
55  	public void error(String message, String sourceName, int line, String lineSource, int lineOffset) {
56  		
57  		// Only log the first error...
58  		if(this.errorLine < 1) {
59  			this.errorLine = line;
60  			this.yuiErrorMessage = message;
61  		}
62  		else if(this.errorLine == line) {
63  			this.yuiErrorMessage += "] [" + message;
64  		}
65  	}
66  
67  	
68  	/***
69  	 * Creates an EvaluatorException that will be thrown by Rhino. 
70  	 * 
71  	 * @param message a String describing the error
72  	 * @param sourceName a String describing the JavaScript source where the error occured; typically a filename or URL
73  	 * @param line the line number associated with the error
74  	 * @param lineSource the text of the line (may be null)
75  	 * @param lineOffset the offset into lineSource where problem was detected
76  	 * 
77  	 * @return an EvaluatorException that will be thrown.
78  	 */
79  	public EvaluatorException runtimeError(String message, String sourceName, int line, String lineSource, int lineOffset) {
80  		StringBuffer errorMsg = new StringBuffer("YUI failed to minify the bundle with id: '" + status.getCurrentBundle().getName() + "'.\n");
81  		errorMsg.append("YUI error message(s):[").append(this.yuiErrorMessage).append("]\n");
82  		errorMsg.append("The error happened at this point in your javascript: \n");
83  		errorMsg.append("_______________________________________________\n...\n");
84  		
85  		BufferedReader rd = new BufferedReader(new StringReader(bundleData.toString()));
86  		String s;
87  		int totalLines = 0;
88  		int start = this.errorLine - 10;
89  		
90  		try {
91  			while((s=rd.readLine())!=null){
92  				totalLines++;
93  				if(totalLines >= start && totalLines <= this.errorLine){
94  					errorMsg.append(s);
95  					if(totalLines == this.errorLine)
96  						errorMsg.append(" <-- ERROR");
97  					errorMsg.append("\n");
98  				}
99  			}
100 		} catch (IOException e) {
101 			errorMsg.append("[Jawr suffered an IOException while attempting to show the faulty script]");
102 		}
103 		
104 		errorMsg.append("_______________________________________________");
105 		errorMsg.append("\nIf you can't find the error, try to check the scripts using JSLint (http://www.jslint.com/) to find the conflicting part of the code.\n");
106 
107 		return new EvaluatorException(errorMsg.toString(),status.getCurrentBundle().getName(),this.errorLine);
108 	}
109 
110 	/* (non-Javadoc)
111 	 * @see org.mozilla.javascript.ErrorReporter#warning(java.lang.String, java.lang.String, int, java.lang.String, int)
112 	 */
113 	public void warning(String message, String sourceName, int line, String lineSource, int lineOffset) {
114 		if(log.isDebugEnabled())
115 			log.debug(message);
116 	}
117 
118 }
