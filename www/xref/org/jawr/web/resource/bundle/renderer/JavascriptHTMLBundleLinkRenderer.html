14  package org.jawr.web.resource.bundle.renderer;
15  
16  import org.jawr.web.resource.bundle.factory.util.PathNormalizer;
17  import org.jawr.web.resource.bundle.handler.ResourceBundlesHandler;
18  
19  /***
20   * Renderer that creates javascript link tags. 
21   * 
22   * @author Jordi Hernández Sellés
23   */
24  public class JavascriptHTMLBundleLinkRenderer extends AbstractBundleLinkRenderer implements BundleRenderer{
25      
26      private static final String PRE_TAG = "<script type=\"text/javascript\" src=\"";
27      private static final String POST_TAG = "\" ></script>\n";
28      
29      /*** Creates a new instance of JavascriptHTMLBundleLinkRenderer */
30      public JavascriptHTMLBundleLinkRenderer(ResourceBundlesHandler bundler) {
31          super(bundler);
32      }
33  
34      /* (non-Javadoc)
35       * @see org.jawr.web.resource.bundle.renderer.AbstractBundleLinkRenderer#createBundleLink(java.lang.String, java.lang.String)
36       */
37      protected String createBundleLink(String bundleId, String contextPath) {
38      	StringBuilder sb = new StringBuilder(PRE_TAG);
39          String fullPath = PathNormalizer.joinPaths(getBundler().getConfig().getServletMapping(), bundleId);
40          fullPath = PathNormalizer.joinPaths(contextPath,fullPath);
41  		sb.append(fullPath)
42  			.append(POST_TAG); 
43          return sb.toString();
44      }
45      
46  }
