14  package org.jawr.web.resource.bundle.renderer;
15  
16  import java.io.IOException;
17  import java.io.Writer;
18  import java.util.Iterator;
19  import java.util.List;
20  import java.util.Random;
21  import java.util.Set;
22  
23  import org.jawr.web.resource.bundle.handler.ResourceBundlesHandler;
24  
25  /***
26   * Abstract base class for implementations of a link renderer. 
27   * 
28   * @author Jordi Hernández Sellés
29   */
30  public abstract class AbstractBundleLinkRenderer implements BundleRenderer {
31      
32      private ResourceBundlesHandler bundler;
33      
34      /*** Creates a new instance of AbstractBundleLinkRenderer
35       * @param bundler ResourceBundlesHandler Handles resolving of paths. 
36       */
37      protected AbstractBundleLinkRenderer(ResourceBundlesHandler bundler) {
38          this.bundler = bundler;
39      }
40      
41      /***
42       * Render a link or group of links to the specified resource, included global bundles if applies. 
43       * @param requestedPath String Path that identifies a resource bundle id or one of its members. 
44       * @param contextPath String The context path to prepend to the URL. 
45       * @param includedBundles Set A set of names previously added. None of them will be written out as links to avoid duplication. 
46       * @param useGzip boolean If true, the gzip prefix is added to the URLs so the link will point to the gzipped version. 
47       * @param out Writer Writer to output the tags, typically a JSPWriter.
48       * @throws java.io.IOException 
49       */
50      public void renderBundleLinks(  String requestedPath,
51                                      String contextPath,
52                                      final Set includedBundles, 
53                                      boolean useGzip, 
54                                      Writer out ) throws IOException {
55          
56      	boolean debugOn = bundler.getConfig().isDebugModeOn();
57          String bundleId = bundler.resolveBundleForPath(requestedPath);
58  		
59          // Retrieve the name or names of bundle(s) that belong to/with the requested path. 
60          List resourceNames = bundler.getBundlePaths(bundleId);
61  
62          if( debugOn ) {
63                  addComment("Start adding members resolved by '" + requestedPath + "'. Bundle id is: '" + bundleId + "'",out);
64          }
65  
66          // Add resources to the page as links. 
67          for(Iterator it = resourceNames.iterator();it.hasNext();)
68          {
69                  String resourceName = (String) it.next();
70  
71                  // If the resource had been added before, it will not be included again. 
72                  if(includedBundles.add(resourceName)){
73                  		// In debug mode, all the resources are included separately and use a random parameter to avoid caching. 
74                          if( debugOn ) {
75                                  int random = new Random().nextInt();
76                                  if(random < 0)
77                                  	random*=-1;
78                                  out.write(createBundleLink(resourceName + "?d=" + random,contextPath));
79                          }
80                          else if(useGzip)
81                              out.write(createGzipBundleLink(resourceName,contextPath));
82                          else 
83                              out.write(createBundleLink(resourceName,contextPath));
84                  }
85                  else if( debugOn ) {
86                          addComment("Skipping resource '" + resourceName +"' since it is already included in the page.",out);
87                  }
88          }
89          if( debugOn ) {
90                  addComment("Finished adding members resolved by " + requestedPath,out);
91          }
92      }
93      
94      
95      
96      /***
97       * Adds an HTML comment to the output stream. 
98       * @param commentText
99       * @param out Writer
100      * @throws IOException
101      */
102     protected final void addComment(String commentText, Writer out)  throws IOException {
103     		StringBuilder sb = new StringBuilder("<!-- ");
104             sb.append(commentText).append(" -->").append("\n");
105             out.write(sb.toString());
106     }
107     
108     /***
109      * Creates a link to a bundle in the page, prepending the gzip prefix to its identifier. 
110      * @param resourceName
111      * @param contextPath
112      * @return String
113      */
114     protected String createGzipBundleLink(String resourceName, String contextPath) {
115         // remove '/' from start of name
116         resourceName = resourceName.substring(1,resourceName.length());
117         return createBundleLink(BundleRenderer.GZIP_PATH_PREFIX + resourceName,contextPath);
118     }
119     
120     /***
121      * Creates a link to a bundle in the page, using its identifier. 
122      * @param bundleId
123      * @param contextPath
124      * @return String
125      */
126     protected abstract String createBundleLink(String bundleId, String contextPath);
127 
128     /***
129      * @return ResourceBundlesHandler The resources handler.
130      */
131     public ResourceBundlesHandler getBundler() {
132         return bundler;
133     }
134
