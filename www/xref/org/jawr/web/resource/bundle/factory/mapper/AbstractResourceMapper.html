14  package org.jawr.web.resource.bundle.factory.mapper;
15  
16  import java.util.ArrayList;
17  import java.util.HashMap;
18  import java.util.Iterator;
19  import java.util.List;
20  import java.util.Map;
21  
22  import org.apache.log4j.Logger;
23  import org.jawr.web.exception.DuplicateBundlePathException;
24  import org.jawr.web.resource.ResourceHandler;
25  import org.jawr.web.resource.bundle.JoinableResourceBundle;
26  import org.jawr.web.resource.bundle.factory.util.PathNormalizer;
27  
28  /***
29   * Base class to implement map-based automatic bundles generators. 
30   * The generated bundles are added to a Map instance in which the keys 
31   * are bundles ids and the values are the bundles. 
32   * 
33   * @author Jordi Hernández Sellés
34   *
35   */
36  public abstract class AbstractResourceMapper {
37  	private static final Logger log = Logger.getLogger(AbstractResourceMapper.class);
38  	protected String baseDir;
39  	protected ResourceHandler rsHandler;
40  	protected List currentBundles;
41  	protected String resourceExtension;
42  	private Map bundleMapping;
43  	
44  	public AbstractResourceMapper(String baseDir,
45  			ResourceHandler rsHandler, List currentBundles,
46  			String resourceExtension) {
47  		super();
48  		this.baseDir = PathNormalizer.normalizePath(baseDir);
49  		this.rsHandler = rsHandler;
50  		this.currentBundles = new ArrayList();
51  		if(null != currentBundles)
52  			this.currentBundles.addAll(currentBundles);
53  		this.resourceExtension = resourceExtension;
54  		this.bundleMapping = new HashMap();
55  	}
56  	
57  
58  	/***
59  	 * Find the required files to add to the mapping. Subclasses must use the addBundleToMap method. 
60  	 * @throws DuplicateBundlePathException
61  	 */
62  	protected abstract void addBundlesToMapping() throws DuplicateBundlePathException;
63  	
64  	public final Map getBundleMapping() throws DuplicateBundlePathException{
65  		addBundlesToMapping();
66  		return bundleMapping;
67  	}
68  	
69  	/***
70  	 * Add a bundle and its mapping to the resulting Map. 
71  	 * @param bundleId
72  	 * @param mapping
73  	 * @throws DuplicateBundlePathException 
74  	 */
75  	protected final void addBundleToMap(String bundleId, String mapping) throws DuplicateBundlePathException {
76  		
77  		for(Iterator it = currentBundles.iterator();it.hasNext(); ) {
78  			JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();			
79  			if(bundleId.equals(bundle.getName()) || this.bundleMapping.containsKey(bundleId)){
80  				log.fatal("Duplicate bundle id resulted from mapping:" + bundleId);
81  				throw new DuplicateBundlePathException(bundleId);
82  			}
83  		}
84  		
85  		bundleMapping.put(bundleId, mapping);
86  	}
87  	
88  	
89  }
