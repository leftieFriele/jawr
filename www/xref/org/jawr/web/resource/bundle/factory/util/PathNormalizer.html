14  package org.jawr.web.resource.bundle.factory.util;
15  
16  import java.util.HashSet;
17  import java.util.Iterator;
18  import java.util.Set;
19  import java.util.StringTokenizer;
20  
21  /***
22   * Utility class to work with relative paths. 
23   * 
24   * @author Jordi Hernández Sellés
25   *
26   *
27   */
28  public class PathNormalizer {
29  	private static final String SEPARATOR = "/";
30  	
31  	/***
32  	 * Normalizes a bundle path mapping. If it ends with a wildcard, the wildcard is removed. 
33  	 * @param pathMapping
34  	 * @return
35  	 */
36  	public static final String normalizePathMapping(String pathMapping) {
37  		pathMapping = normalizePath(pathMapping);
38  		if(pathMapping.endsWith("/**"))
39  			pathMapping = pathMapping.substring(0,pathMapping.length()-3);
40  		return pathMapping;
41  	}
42  	
43  	/***
44  	 * Normalizes a path and adds a separator at its start. 
45  	 * @param path
46  	 * @return
47  	 */
48  	public static final String asPath(String path) {
49  		return(SEPARATOR + normalizePath(path));
50  	}
51  	
52  	/***
53  	 * Normalizes two paths and joins them as a single path. 
54  	 * @param prefix
55  	 * @param path
56  	 * @return
57  	 */
58  	public static final String joinPaths(String prefix,String path) {
59  		prefix = PathNormalizer.normalizePath(prefix);
60  		path = PathNormalizer.normalizePath(path);
61  		StringBuilder sb = new StringBuilder(SEPARATOR);
62  		if(!"".equals(prefix))
63  			sb.append(prefix).append(SEPARATOR);
64  		sb.append(path);
65  		
66  		return sb.toString();
67  		
68  	}
69  	
70  	/***
71  	 * Removes leading and trailing separators from a path, and removes 
72  	 * double separators (// is replaced by /). 
73  	 * @param path
74  	 * @return
75  	 */
76  	public static final String normalizePath(String path) {
77  		path = path.replaceAll("//", SEPARATOR);
78  		StringTokenizer tk = new StringTokenizer(path,SEPARATOR);
79  		StringBuilder sb = new StringBuilder();
80  		while(tk.hasMoreTokens()) {
81  			sb.append(tk.nextToken());
82  			if(tk.hasMoreTokens())
83  				sb.append(SEPARATOR);
84  		}
85  		return sb.toString();
86  		
87  	}
88  	
89  	/***
90  	 * Normalizes all the paths in a Set. 
91  	 * @param paths
92  	 * @return
93  	 */
94  	public static final Set normalizePaths(Set paths) {
95  		Set ret = new HashSet();
96  		for(Iterator it = paths.iterator();it.hasNext();) {
97  			String path = normalizePath((String)it.next());
98  			ret.add(path);
99  		}
100 		return ret;
101 	}
102 }
