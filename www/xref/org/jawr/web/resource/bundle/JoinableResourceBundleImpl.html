96       public JoinableResourceBundleImpl(	String name, 
97          									String fileExtension,
98          									InclusionPattern inclusionPattern,
99          									List pathMappings, 
100 											ResourceHandler resourceHandler,
101                                             String urlPrefix) {
102 		this(name, fileExtension, inclusionPattern, resourceHandler, urlPrefix);
103       
104 		if(log.isInfoEnabled())
105 			log.info("Adding mapped files for bundle " + getName());
106 		this.pathMappings = pathMappings;
107 		
108 		initPathList();
109 		if(log.isInfoEnabled())
110 			log.info("Added " + this.itemPathList .size() + " files and " + licensesPathList.size() + " licenses for the bundle " + getName());
111 		
112 	}

116 	/***
117 	 * Detects all files that belong to this bundle and adds them to the 
118 	 * items path list. 
119 	 */
120 	private void initPathList()	{
121 		if(log.isDebugEnabled())
122 			log.debug("Creating bundle path List for " + getName());
123 		
124 		for(Iterator it = pathMappings.iterator();it.hasNext();)
125 		{
126 			String pathMapping = (String) it.next();
127 
128 			// path ends in /, the folder is included without subfolders
129 			if(pathMapping.endsWith("/")) {
130 				addItemsFromDir(pathMapping,false);
131 			}
132 			// path ends in /, the folder is included with all subfolders
133 			else if(pathMapping.endsWith("/**")){
134 				addItemsFromDir(pathMapping.substring(0,pathMapping.lastIndexOf("**")),true);
135 			}
136 			else if(pathMapping.endsWith(fileExtension)){
137 				itemPathList.add(PathNormalizer.asPath(pathMapping));
138 			}
139 			else if(pathMapping.endsWith(LICENSES_FILENAME)){
140 				licensesPathList.add(PathNormalizer.asPath(pathMapping));
141 			}
142 			else log.warn("Wrong mapping [" + pathMapping  
143 							+ "] for bundle [" 
144 							+ this.name 
145 							+ "]. Please check configuration. ");
146 		}
147 		if(log.isDebugEnabled())
148 			log.debug("Finished creating bundle path List for " + getName());
149 	}

151 	/***
152 	 * Adds all the resources within a path to the item path list. 
153 	 * @param dirName
154 	 * @param addSubDirs boolean If subfolders will be included. In such case, 
155 	 * 							 every folder below the path is included.  
156 	 */
157 	private void addItemsFromDir(String dirName, boolean addSubDirs)
158 	{
159 		Set resources = resourceHandler.getResourceNames(dirName);
160 		
161 			
162 		if(log.isDebugEnabled()) {
163 			log.debug("Adding " + resources.size() + " resources from path [" + dirName + "] to bundle " + getName());
164 		}
165 
166 		
167 		// If the directory contains a sorting file, it is used to order the resources. 
168 		if(resources.contains(SORT_FILE_NAME) || resources.contains("/" + SORT_FILE_NAME)) {
169 			SortFileParser parser;
170 			try {
171 				String sortFilePath = PathNormalizer.joinPaths(dirName, SORT_FILE_NAME);
172 				parser = new SortFileParser(resourceHandler.getResource(sortFilePath),
173 															resources,
174 															dirName);
175 			} catch (ResourceNotFoundException e) {
176 				throw new RuntimeException("Unexpected ResourceNotFoundException when reading a sorting file[" + e.getRequestedPath() + "]",e);
177 			}
178 			List sortedResources = parser.getSortedResources();
179 			for(Iterator it = sortedResources.iterator();it.hasNext();) {
180 				String resourceName = (String) it.next();
181 				
182 				// Add subfolders or files
183 				if(resourceName.endsWith(fileExtension)){
184 					itemPathList.add(PathNormalizer.asPath(resourceName));
185 					if(log.isDebugEnabled())
186 						log.debug("Added to item path list from the sorting file:" + resourceName);
187 				}
188 				else if(addSubDirs && resourceHandler.isDirectory(resourceName))
189 					addItemsFromDir(resourceName,true);
190 			}		
191 		}
192 		
193 		// Add licenses file
194 		if(resources.contains(LICENSES_FILENAME) || resources.contains("/" + LICENSES_FILENAME)) {
195 			licensesPathList.add(PathNormalizer.joinPaths(dirName, LICENSES_FILENAME));
196 		}
197 		
198 		// Add remaining resources (remaining after sorting, or all if no sort file present)
199 		List folders = new ArrayList();
200 		for(Iterator it = resources.iterator();it.hasNext();) {
201 			String resourceName = (String) it.next();
202 			String resourcePath = PathNormalizer.joinPaths(dirName, resourceName);
203 			if(resourceName.endsWith(fileExtension)) {
204 				itemPathList.add(PathNormalizer.asPath(resourcePath));
205 
206 				if(log.isDebugEnabled())
207 					log.debug("Added to item path list:" + PathNormalizer.asPath(resourcePath));
208 			}
209 			else if(addSubDirs && resourceHandler.isDirectory(resourcePath))
210 				folders.add(resourceName);
211 		}
212 		
213 		// Add subfolders if requested. Subfolders are added last unless specified in sorting file. 
214 		if(addSubDirs)
215 		{
216 			for(Iterator it = folders.iterator();it.hasNext();) {
217 				String folderName = (String) it.next();
218 				addItemsFromDir(PathNormalizer.joinPaths(dirName, folderName),true);
219 			}
220 		}
221 	}
