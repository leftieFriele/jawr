14  package org.jawr.web.resource;
15  
16  import java.io.File;
17  import java.io.InputStream;
18  import java.io.InputStreamReader;
19  import java.io.Reader;
20  import java.nio.charset.Charset;
21  import java.util.HashSet;
22  import java.util.Iterator;
23  import java.util.Set;
24  
25  import javax.servlet.ServletContext;
26  
27  import org.jawr.web.exception.ResourceNotFoundException;
28  
29  /***
30   * Implementation of resourcehandler that gets its resources
31   * from a ServletContext instance, as resources contained in 
32   * a deployed war archive (or exploded directory). The generated
33   * files are stored in the web server temporary folder. 
34   * 
35   * @author Jordi Hernández Sellés
36   *
37   */
38  public class ServletContextResourceHandler extends AbstractResourceHandler implements ResourceHandler {
39  	
40  	private ServletContext context;
41  	private static final String SERVLET_CONTEXT_TEMPDIR = "javax.servlet.context.tempdir";
42  	
43  	/***
44  	 * @param resourceDir
45  	 * @param context
46  	 */
47  	public ServletContextResourceHandler(ServletContext context, Charset charset) {
48  		super((File) context.getAttribute(SERVLET_CONTEXT_TEMPDIR),charset);		
49  		this.context  = context;
50  		this.charset = charset;
51  	}
52  
53  	/* (non-Javadoc)
54  	 * @see org.jawr.web.resource.bundle.ResourceHandler#getResource(java.lang.String)
55  	 */
56  	public Reader getResource(String resourceName)  throws ResourceNotFoundException{
57  		InputStream is = context.getResourceAsStream(resourceName);		
58  		if(null == is)
59  			throw new ResourceNotFoundException(resourceName);
60  		
61  		return new InputStreamReader(is, charset);
62  	}
63  
64  	/* (non-Javadoc)
65  	 * @see org.jawr.web.resource.bundle.ResourceHandler#getResourceNames(java.lang.String)
66  	 */
67  	public Set getResourceNames(String path) {
68  		Set paths = context.getResourcePaths(path);
69  		Set names = new HashSet();
70  		int length = path.length();
71  		for(Iterator it = paths.iterator();it.hasNext();) {
72  			String resourcePath = (String) it.next();
73  			names.add(resourcePath.substring(length, resourcePath.length()));
74  		}
75  		return names;
76  	}
77  
78  	/* (non-Javadoc)
79  	 * @see org.jawr.web.resource.bundle.ResourceHandler#isDirectory(java.lang.String)
80  	 */
81  	public boolean isDirectory(String path) {
82  		Set paths = context.getResourcePaths(path);
83  		return (null != paths && paths.size() > 0);
84  	}
85  
86  }
