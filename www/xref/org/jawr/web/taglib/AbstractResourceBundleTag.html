29  /***
30   * Abstract implementation of a tag lib component which will retrieve a Jawr config
31   * object from the servlet context and use it to render bundles of resources according 
32   * to its src attribute.  
33   * 
34   * @author Jordi Hernández Sellés
35   *
36   */
37  public abstract class AbstractResourceBundleTag extends TagSupport {
38  	
39  	private static final Logger log = Logger.getLogger(AbstractResourceBundleTag.class.getName());
40  	private static final String ADDED_COLLECTIONS_LOG = "org.jawr.web.taglib.ADDED_COLLECTIONS_LOG";
41  	
42  	private String src;
43          protected BundleRenderer renderer;
44          
45  
46  
47  	/* (non-Javadoc)
48  	 * @see javax.servlet.jsp.tagext.TagSupport#doStartTag()
49  	 */
50  	public int doStartTag() throws JspException {		
51              
52              // In some cases, the container will reuse the tag instance so the renderer can be reused. 
53              // The spec says if properties are different then new instances will be created, otherwise instances are reused.  
54              if(null == this.renderer)
55                  this.renderer = createRenderer();		
56              
57              try {
58                  HttpServletRequest request = (HttpServletRequest)pageContext.getRequest();
59                  renderer.renderBundleLinks( src,
60                                              pageContext.getServletContext().getContextPath(),
61                                              getAddedBundlesLog(request),
62                                              shouldUseGZIP(request),
63                                              pageContext.getOut());
64              } catch (IOException ex) {
65                  throw new JspException("Unexpected IOException when writing script tags for path " + src,ex);
66              }
67  
68              return super.doStartTag();
69  	}
70  	
71  	/***
72  	 * Retrieve or create the set to store all added bundles, which is used
73  	 * to avoid adding a bundle more than once during a single request. 
74  	 * @param request
75  	 * @return
76  	 */
77  	private Set getAddedBundlesLog(ServletRequest request)
78  	{
79  		Set set = null;
80  		if(null != request.getAttribute(ADDED_COLLECTIONS_LOG))
81  			set = (Set) request.getAttribute(ADDED_COLLECTIONS_LOG);
82  		else
83  		{
84  			set = new HashSet();
85  			request.setAttribute(ADDED_COLLECTIONS_LOG, set);
86  		}
87  		return set;
88  	}
89          
90          /***
91           * Determines wether gzip is suitable for the current request. 
92           * @param req 
93           * @return 
94           */
95          private boolean shouldUseGZIP(HttpServletRequest req) {
96              boolean rets;
97              JawrConfig jeesConfig = renderer.getBundler().getConfig();
98              // If gzip is completely off, return false. 
99              if(!jeesConfig.isGzipResourcesModeOn())
100                 rets = false;
101             else if(req.getHeader("Accept-Encoding") != null && 
102 		    req.getHeader("Accept-Encoding").indexOf("gzip") != -1 ) {
103                 
104                 // If gzip for IE6 or less is off, the user agent is checked to avoid compression. 
105                 if(!jeesConfig.isGzipResourcesForIESixOn()) {
106                     String agent = req.getHeader("User-Agent");
107                     if(log.isInfoEnabled())
108                         log.info("User-Agent for this request:" +agent);                    
109                     
110                     if(null != agent && agent.indexOf("MSIE") != -1) {
111                         rets =  agent.indexOf("MSIE 4") == -1 && 
112                                 agent.indexOf("MSIE 5") == -1 && 
113                                 agent.indexOf("MSIE 6") == -1;
114                         if(log.isInfoEnabled())
115                             log.info("Gzip enablement for IE executed, with result:" + rets);
116                     }
117                     else rets = true;
118                 }
119                 else rets = true;                
120             }
121             else rets = false;
122             
123             return rets;
124         }

139 	protected abstract BundleRenderer createRenderer();
