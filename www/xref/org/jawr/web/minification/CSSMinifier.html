14  package org.jawr.web.minification;
15  
16  import java.util.regex.Matcher;
17  import java.util.regex.Pattern;
18  
19  /***
20   * Minifies CSS files by removing expendable whitespace and comments. 
21   * 
22   * @author Jordi Hernández Sellés
23   *
24   */
25  public class CSSMinifier {
26  	
27  	// This regex captures comments, plus newlines and tabs
28  	private static final String commentRegex ="(///*[^*]*//*+([^/][^*]*//*+)*/)|//r|//n|//t|//f";
29  	
30  	// This regex captures, in order: single or double quote enclosed content, 
31  	// brackets, parentheses,colons and semicolons and any spaces around them, and occurrences of one or more spaces. 
32  	private static final String spacesRegex = "('.*')|(\".*\")|(//s*//{//s*)|(//s*//}//s*)|(//s*//(//s*)|(//s*//)//s*)|(//s*;//s*)|(//s*://s*)|( +)";
33  	
34  	private static final Pattern cssPattern = Pattern.compile(commentRegex, Pattern.DOTALL);
35  	private static final Pattern spacesPattern = Pattern.compile(spacesRegex, Pattern.DOTALL);
36  	
37  	private static final String SPACE = " ";
38  	private static final String QUOTE = "'";
39  	private static final String DQUOTE = "\"";
40  	private static final String BRACKET_OPEN = "{";
41  	private static final String BRACKET_CLOSE = "}";
42  	private static final String PAREN_OPEN = "(";
43  	private static final String PAREN_CLOSE = ")";
44  
45  	private static final String COLON = ":";
46  	private static final String SEMICOLON = ";";
47  
48  	/***
49  	 * @param data CSS to minify
50  	 * @return StringBuffer Minified CSS. 
51  	 */
52  	public StringBuffer minifyCSS(StringBuffer data) {
53  		// Remove comments and carriage returns
54  		String noComments = cssPattern.matcher(data.toString()).replaceAll("");
55  		
56  		// Match all empty space we can minify 
57  		Matcher matcher = spacesPattern.matcher(noComments);
58  		StringBuffer sb = new StringBuffer();
59  		while(matcher.find()) {
60  			String replacement = SPACE;
61  			String match = matcher.group();
62  			
63  			// No replacement for strings
64  			if(match.indexOf(QUOTE) != -1 || match.indexOf(DQUOTE) != -1)
65  				replacement = match;
66  			else if(match.indexOf(BRACKET_OPEN) != -1)
67  				replacement = BRACKET_OPEN;
68  			else if(match.indexOf(BRACKET_CLOSE) != -1)
69  				replacement = BRACKET_CLOSE;
70  			else if(match.indexOf(PAREN_OPEN) != -1)
71  				replacement = PAREN_OPEN;
72  			else if(match.indexOf(PAREN_CLOSE) != -1)
73  				replacement = PAREN_CLOSE;
74  			else if(match.indexOf(COLON) != -1)
75  				replacement = COLON;
76  			else if(match.indexOf(SEMICOLON) != -1)
77  				replacement = SEMICOLON;
78  			
79  			matcher.appendReplacement(sb, replacement);
80  		}
81  		matcher.appendTail(sb);
82  		return sb;
83  	}
84  
85  }
