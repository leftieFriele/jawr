66  	/***
67  	 * Test the ability to include license files in a bundle. 
68  	 */
69  	public void testDoPostProcessBundle() {
70  	try {
71  	    List cols = new ArrayList();
72  	    cols.add(resourcebundle);
73  	    ResourceBundlesHandler collector = new ResourceBundlesHandlerImpl(cols, rsHandler, jeesConfig);
74  	    collector.initAllBundles();
75  	    ByteArrayOutputStream baOs = new ByteArrayOutputStream();
76  	    WritableByteChannel wrChannel = Channels.newChannel(baOs);
77  	    Writer writer = Channels.newWriter(wrChannel, jeesConfig.getResourceCharset().name());
78  	    collector.writeBundleTo("/js/script.js", writer);
79  	    BundleProcessingStatus status = new BundleProcessingStatus(resourcebundle,rsHandler,jeesConfig);
80  	    StringBuffer sb = processor.postProcessBundle(status, new StringBuffer(baOs.toString(jeesConfig.getResourceCharset().name())));
81  	    String contents = sb.toString();
82  	    assertTrue("License in root folder not included",(contents.indexOf("/** License in folder **/")!=-1));
83  	    assertTrue("License in subfolder not included",(contents.indexOf("/** License in subfolder **/")!=-1));
84  	} catch (Exception ex) {
85  	    fail("Error:" +ex.getMessage());
86  	}
