14  package net.jawr.web.taglib;

16  import java.io.IOException;

18  import javax.servlet.http.HttpServletRequest;
19  import javax.servlet.jsp.JspException;
20  import javax.servlet.jsp.tagext.TagSupport;

22  import net.jawr.web.config.JawrConfig;
23  import net.jawr.web.resource.bundle.renderer.BundleRenderer;

26  /***
27   * Abstract implementation of a tag lib component which will retrieve a Jawr config
28   * object from the servlet context and use it to render bundles of resources according 
29   * to its src attribute.  
30   * 
31   * @author Jordi Hernández Sellés
32   *
33   */
34  public abstract class AbstractResourceBundleTag extends TagSupport {
35  	
36  	//private static final Logger log = Logger.getLogger(AbstractResourceBundleTag.class.getName());
37  	
38  	private String src;
39  	protected BundleRenderer renderer;
40          
41  
42  
43  	/* (non-Javadoc)
44  	 * @see javax.servlet.jsp.tagext.TagSupport#doStartTag()
45  	 */
46  	public int doStartTag() throws JspException {		
47              
48              // In some cases, the container will reuse the tag instance so the renderer can be reused. 
49              // The spec says if properties are different then new instances will be created, otherwise instances are reused.  
50              if(null == this.renderer)
51                  this.renderer = createRenderer();		
52              
53              try {
54                  HttpServletRequest request = (HttpServletRequest)pageContext.getRequest();
55                  renderer.renderBundleLinks( src,
56                                              request.getContextPath(),
57                                              RendererRequestUtils.getAddedBundlesLog(request),
58                                              shouldUseGZIP(request),
59                                              pageContext.getOut());
60              } catch (IOException ex) {
61                  throw new JspException("Unexpected IOException when writing script tags for path " + src,ex);
62              }
63  
64              return super.doStartTag();
65  	}
66          
67      /***
68       * Determines wether gzip is suitable for the current request. 
69       * @param req 
70       * @return 
71       */
72      private boolean shouldUseGZIP(HttpServletRequest req) {
73          boolean rets;
74          JawrConfig jeesConfig = renderer.getBundler().getConfig();
75          rets = RendererRequestUtils.isRequestGzippable(req, jeesConfig);
76          return rets;
77      }
78  
79  	
80  	/***
81  	 * Set the source of the resource or collection to retrieve. 
82  	 * @param src
83  	 */
84  	public void setSrc(String src) {
85  		this.src = src;
86  	}
87  	
88  	/***
89  	 * Retrieve the ResourceCollector from context. Each implementation will use a different key
90  	 * to retrieve it. 
91  	 * @return
92  	 */
93  	protected abstract BundleRenderer createRenderer();
94  }
