68  	public JSMin(InputStream in, OutputStream out) {
69  		this.in = new PushbackInputStream(in);
70  		this.out = out;

73  	/***
74  	 * isAlphanum -- return true if the character is a letter, digit,
75  	 * underscore, dollar sign, or non-ASCII character.
76  	 */
77  	static boolean isAlphanum(int c) {
78  		return ( (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') || 
79  				 (c >= 'A' && c <= 'Z') || c == '_' || c == '$' || c == '//' || 
80  				 c > 126);
81  	}
82  
83  	/***
84  	 * get -- return the next character from stdin. Watch out for lookahead. If
85  	 * the character is a control character, translate it to a space or
86  	 * linefeed.
87  	 */
88  	int get() throws IOException {
89  		int c = in.read();
90  
91  		if (c >= ' ' || c == '\n' || c == EOF) {
92  			return c;
93  		}
94  
95  		if (c == '\r') {
96  			return '\n';
97  		}
98  		
99  		return ' ';
100 	}

104 	/***
105 	 * Get the next character without getting it.
106 	 */
107 	int peek() throws IOException {
108 		int lookaheadChar = in.read();
109 		in.unread(lookaheadChar);
110 		return lookaheadChar;
111 	}

113 	/***
114 	 * next -- get the next character, excluding comments. peek() is used to see
115 	 * if a '/' is followed by a '/' or '*'.
116 	 */
117 	int next() throws IOException, UnterminatedCommentException {
118 		int c = get();
119 		if (c == '/') {
120 			switch (peek()) {
121 			case '/':
122 				for (;;) {
123 					c = get();
124 					if (c <= '\n') {
125 						return c;
126 					}
127 				}
128 
129 			case '*':
130 				get();
131 				for (;;) {
132 					switch (get()) {
133 					case '*':
134 						if (peek() == '/') {
135 							get();
136 							return ' ';
137 						}
138 						break;
139 					case EOF:
140 						throw new UnterminatedCommentException();
141 					}
142 				}
143 
144 			default:
145 				return c;
146 			}
147 
148 		}
149 		return c;
150 	}

152 	/***
153 	 * action -- do something! What you do is determined by the argument: 1
154 	 * Output A. Copy B to A. Get the next B. 2 Copy B to A. Get the next B.
155 	 * (Delete A). 3 Get the next B. (Delete B). action treats a string as a
156 	 * single character. Wow! action recognizes a regular expression if it is
157 	 * preceded by ( or , or =.
158 	 */
159 
160 	void action(int d) throws IOException, UnterminatedRegExpLiteralException,
161 			UnterminatedCommentException, UnterminatedStringLiteralException {
162 		switch (d) {
163 		case 1:
164 			out.write(theA);
165 		case 2:
166 			theA = theB;
167 
168 			if (theA == '\'' || theA == '"') {
169 				for (;;) {
170 					out.write(theA);
171 					theA = get();
172 					if (theA == theB) {
173 						break;
174 					}
175 					if (theA <= '\n') {
176 						throw new UnterminatedStringLiteralException();
177 					}
178 					if (theA == '//') {
179 						out.write(theA);
180 						theA = get();
181 					}
182 				}
183 			}
184 			
185 		case 3:
186 			theB = next();
187 			if (theB == '/' && (theA == '(' || theA == ',' || theA == '=' ||
188                     			theA == ':' || theA == '[' || theA == '!' || 
189                     			theA == '&' || theA == '|' || theA == '?' || 
190                     			theA == '{' || theA == '}' || theA == ';' || 
191                     			theA == '\n')) {
192 				out.write(theA);
193 				out.write(theB);
194 				for (;;) {
195 					theA = get();
196 					if (theA == '/') {
197 						break;
198 					} else if (theA == '//') {
199 						out.write(theA);
200 						theA = get();
201 					} else if (theA <= '\n') {
202 						throw new UnterminatedRegExpLiteralException();
203 					}
204 					out.write(theA);
205 				}
206 				theB = next();
207 			}
208 		}
209 	}

211 	/***
212 	 * jsmin -- Copy the input to the output, deleting the characters which are
213 	 * insignificant to JavaScript. Comments will be removed. Tabs will be
214 	 * replaced with spaces. Carriage returns will be replaced with linefeeds.
215 	 * Most spaces and linefeeds will be removed.
216 	 */
217 	public void jsmin() throws IOException, UnterminatedRegExpLiteralException, UnterminatedCommentException, UnterminatedStringLiteralException{
218 		theA = '\n';
219 		action(3);
220 		while (theA != EOF) {
221 			switch (theA) {
222 			case ' ':
223 				if (isAlphanum(theB)) {
224 					action(1);
225 				} else {
226 					action(2);
227 				}
228 				break;
229 			case '\n':
230 				switch (theB) {
231 				case '{':
232 				case '[':
233 				case '(':
234 				case '+':
235 				case '-':
236 					action(1);
237 					break;
238 				case ' ':
239 					action(3);
240 					break;
241 				default:
242 					if (isAlphanum(theB)) {
243 						action(1);
244 					} else {
245 						action(2);
246 					}
247 				}
248 				break;
249 			default:
250 				switch (theB) {
251 				case ' ':
252 					if (isAlphanum(theA)) {
253 						action(1);
254 						break;
255 					}
256 					action(3);
257 					break;
258 				case '\n':
259 					switch (theA) {
260 					case '}':
261 					case ']':
262 					case ')':
263 					case '+':
264 					case '-':
265 					case '"':
266 					case '\'':
267 						action(1);
268 						break;
269 					default:
270 						if (isAlphanum(theA)) {
271 							action(1);
272 						} else {
273 							action(3);
274 						}
275 					}
276 					break;
277 				default:
278 					action(1);
279 					break;
280 				}
281 			}
282 		}
283 		out.flush();
284 	}
