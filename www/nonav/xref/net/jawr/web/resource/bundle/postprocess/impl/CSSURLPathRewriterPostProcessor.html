16  import java.io.IOException;
17  import java.util.ArrayList;
18  import java.util.Iterator;
19  import java.util.List;
20  import java.util.regex.Matcher;
21  import java.util.regex.Pattern;
22  
23  import net.jawr.web.resource.bundle.postprocess.AbstractChainedResourceBundlePostProcessor;
24  import net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus;

26  /***
27   * Single file postprocessor used to rewrite CSS URLs according to the new relative locations of the references when
28   * added to a bundle. Since the path changes, the URLs must be rewritten accordingly.  
29   * URLs in css files are expected to be according to the css spec (see http://www.w3.org/TR/REC-CSS2/syndata.html#value-def-uri). 
30   * Thus, single double, or no quotes enclosing the url are allowed (and remain as they are after rewriting). Escaped parens and quotes 
31   * are allowed within the url.   
32   *  
33   * @author Jordi Hernández Sellés
34   *
35   */
36  public class CSSURLPathRewriterPostProcessor extends
37  		AbstractChainedResourceBundlePostProcessor {
38  	
39  	private static final Pattern urlPattern = Pattern.compile(	"url//(//s*" // 'url(' and any number of whitespaces 
40  																+ "((//////))|[^)])*" // any sequence of characters, except an unescaped ')'
41  																+ "//s*//)",  // Any number of whitespaces, then ')'
42  																Pattern.CASE_INSENSITIVE); // works with 'URL('
43  	private static final String backRefRegex = "(//.//./)";
44  	private static final Pattern backRefRegexPattern = Pattern.compile(backRefRegex);
45  	private static final Pattern URLSeparatorPattern = Pattern.compile("/");
46  	
47  	/* (non-Javadoc)
48  	 * @see net.jawr.web.resource.bundle.postprocess.impl.AbstractChainedResourceBundlePostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus, java.lang.StringBuffer)
49  	 */
50  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status,
51  			StringBuffer bundleData) throws IOException {
52  		
53  		String data = bundleData.toString();
54  		
55  		// Initial backrefs, for prefix. 
56  		int initialBackRefs = 1;
57  		if(! "".equals(status.getJeesConfig().getServletMapping()))
58  			initialBackRefs++;
59  		
60  		String bundleName = status.getCurrentBundle().getName();
61  		initialBackRefs += numberOfForwardReferences(bundleName);
62  		List resourceBackRefs = getPathNamesInResource(status.getLastPathAdded());
63  		Matcher matcher = urlPattern.matcher(data);
64  		StringBuffer sb = new StringBuffer();
65  		while(matcher.find()) {
66  			List backRefs = new ArrayList();
67  			backRefs.addAll(resourceBackRefs);
68  			String url = getUrlPath(matcher.group(), initialBackRefs, backRefs);
69  			matcher.appendReplacement(sb, adaptToMatcher(url));
70  		}
71  		matcher.appendTail(sb);
72  		return sb;
73  	}

76  	/***
77  	 * Transform a matched url so it points to the proper relative path with respect to the given path.  
78  	 * @param matcher
79  	 * @param initialBackRefs
80  	 * @return
81  	 */
82  	private String getUrlPath(String match, int initialBackRefs, List resourceBackRefs) {
83  
84  		String url = match.substring(match.indexOf('(')+1,match.lastIndexOf(')'))
85  					.trim();
86  
87  		// To keep quotes as they are, first they are checked and removed. 
88  		String quoteStr = "";
89  		if( url.startsWith("'") || url.startsWith("\"")) {
90  			quoteStr = url.charAt(0)+"";
91  			url = url.substring(1,url.length()-1);
92  		}
93  		
94  		// Check if the URL is absolute, if it is return it as is. 
95  		int firstSlash = url.indexOf('/');
96  		if(firstSlash != -1 && url.charAt(++firstSlash) == '/'){
97  			StringBuffer sb = new StringBuffer("url(");
98  			sb.append(quoteStr).append(url).append(quoteStr).append(")");
99  			return sb.toString();
100 		}
101 		
102 		int backRefsInURL = 0;
103 		// Remove leading slash
104 		if(url.startsWith("../")) {
105 			Matcher backUrls = backRefRegexPattern.matcher(url);
106 			while(backUrls.find())
107 				backRefsInURL++;
108 			url = url.replaceAll(backRefRegex, "");
109 		}
110 		else if(url.startsWith("/"))
111 			url = url.substring(1,url.length());
112 		else if(url.startsWith("./"))
113 			url = url.substring(2,url.length());
114 		
115 		// Adjust the forward pathnames according to the backrefs in the url
116 		if(backRefsInURL > 0 && resourceBackRefs.size() > 0) {
117 			int size = resourceBackRefs.size();
118 			int counter = 0;
119 			while(size > 0) {
120 				counter++;
121 				if(counter >= 50)
122 					throw new RuntimeException("Infinite loop on url: " + url + " :size:" + size);
123 				size--;
124 				backRefsInURL--;
125 				resourceBackRefs.remove(size);
126 				if(backRefsInURL == 0)
127 					break;
128 			}
129 		}		
130 		
131 		// Start rendering the result, starting by the initial quote, if any. 
132 		StringBuffer urlPrefix = new StringBuffer("url(").append(quoteStr);
133 		
134 		// Add the back references as needed
135 		for (int i = 0; i < initialBackRefs; i++) {
136 			urlPrefix.append("../");
137 		}		
138 		for(Iterator it = resourceBackRefs.iterator();it.hasNext(); ) {
139 			urlPrefix.append(it.next()).append("/");			
140 		}
141 		return urlPrefix.append(url).append(quoteStr).append(")").toString();
142 	}

144 	/***
145 	 * Rewrite a string so java's Matcher won't choke with it, by replacing
146 	 * special conflicting characters. 
147 	 * @param url
148 	 * @return
149 	 */
150 	private String adaptToMatcher(String url) {
151 		// Double the backslashes, so they are left as they are after replacement. 
152 		url = url.replaceAll("////", "////////");
153 		// Add backslashes after dollar signs 
154 		url = url.replaceAll("//$", "//////$");
155 		return url;
156 	}
157 	
158 	/***
159 	 * Gets the path names within a reource URL, exluding the resource file name. 
160 	 * @param resourceURL
161 	 * @return
162 	 */
163 	private List getPathNamesInResource(String resourceURL) {
164 		List names = new ArrayList();
165 		String[] namesArray = resourceURL.split("/");
166 		
167 		// Add all but the last pathname
168 		for (int i = 0; i < namesArray.length -1 ; i++) {
169 			if(!"".equals(namesArray[i]))
170 				names.add(namesArray[i]);
171 		}
172 		
173 		return names;
174 	}
175 	
176 	/***
177 	 * Find the number of occurrences of / within a url. 
178 	 * @param url
179 	 * @return
180 	 */
181 	private int numberOfForwardReferences(String url) {
182 		Matcher backUrls = URLSeparatorPattern.matcher(url);
183 		int numMatches = 0;
184 		while(backUrls.find())
185 			numMatches++;
186 		
187 		if(url.startsWith("/"))
188 			numMatches--;
189 		
190 		return numMatches;
191 	}
