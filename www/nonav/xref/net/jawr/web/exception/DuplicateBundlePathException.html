14  package net.jawr.web.exception;
15  
16  /***
17   * Exception to throw during configuration process, if the mappings result in a duplicate mapping for a given path.
18   *  
19   * @author Jordi Hernández Sellés
20   */
21  public class DuplicateBundlePathException extends Exception {
22  	private static final long serialVersionUID = 1848915729978060353L;
23  	private String bundlePath;
24  
25  	public DuplicateBundlePathException(String bundleName) {
26  		super();
27  		this.bundlePath = bundleName;
28  	}
29  
30  	public String getBundlePath() {
31  		return bundlePath;
32  	}
33  
34  	public String getMessage() {
35  		return "At least two bundles share the pathname: " + getBundlePath();
36  	}
37  }
